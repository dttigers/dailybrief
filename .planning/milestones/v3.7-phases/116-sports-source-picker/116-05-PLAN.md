    ```typescript
    /** Mirror of vigil-core's League union. */
    export type League = 'mlb' | 'nfl' | 'nba' | 'nhl'

    /** Mirror of vigil-core's SportsSelections shape (D-02 / D-05). */
    export interface SportsSelections {
      enabledLeagues: League[]
      favoriteTeams: {
        mlb?: string
        nfl?: string
        nba?: string
        nhl?: string
      }
    }

    /** Mirror of vigil-core's TeamListEntry shape (D-05: id is BDL team_id as string). */
    export interface TeamListEntry {
      id: string
      name: string
    }

    /**
     * Fetches the user's persisted sports picker selections.
     * Server returns the empty default { enabledLeagues: [], favoriteTeams: {} }
     * when no row exists (D-10) — the caller does NOT need to handle 404.
     */
    export async function getSportsSelections(): Promise<SportsSelections> {
      const res = await vigilFetch('/v1/sports/selections')
      if (!res.ok) throw new Error(`Failed to fetch sports selections: ${res.status}`)
      return (await res.json()) as SportsSelections
    }

    /**
     * Persists the user's sports picker selections. Server overwrites wholesale (D-03).
     * Empty default { enabledLeagues: [], favoriteTeams: {} } IS valid input.
     * Validation (max 4 enabled, allowlisted league keys, string team values) enforced server-side.
     */
    export async function setSportsSelections(s: SportsSelections): Promise<void> {
      const res = await vigilFetch('/v1/sports/selections', {
        method: 'PUT',
        body: JSON.stringify(s),
      })
      if (!res.ok) throw new Error(`Failed to save sports selections: ${res.status}`)
    }

    /**
     * Fetches the team roster for a league for the picker dropdown.
     * Server-side cached for 24h (D-07); client-side caching handled by the caller's component state.
     * Unwraps the { teams } envelope — caller gets the array directly.
     */
    export async function getSportsTeams(league: League): Promise<TeamListEntry[]> {
      const res = await vigilFetch(`/v1/sports/teams/${league}`)
      if (!res.ok) throw new Error(`Failed to fetch teams for ${league}: ${res.status}`)
      const body = (await res.json()) as { teams: TeamListEntry[] }
      return body.teams
    }
    ```

## STRIDE Threat Register (ASVS L1)

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-116-05-01 | Spoofing | Cross-site request forging PUT /v1/sports/selections | mitigate | The PUT goes through `vigilFetch` which attaches `Authorization: Bearer <jwt>` from sessionStorage (vigil-pwa/src/api/client.ts:53-58). Cross-site requests CANNOT read sessionStorage and CANNOT attach the Authorization header. CSRF is structurally impossible — no cookies involved. Verify: `grep -n "Authorization.*Bearer" vigil-pwa/src/api/client.ts`. |
| T-116-05-02 | Information Disclosure | Team names rendered as raw HTML (XSS) via `team.name` from BDL | mitigate | All team names are rendered via React text nodes (`{team.name}`, `{LEAGUE_LABELS[league]}`). React escapes by default — no `dangerouslySetInnerHTML`, no `eval`. Plan 02's server-side normalization picks `display_name` or `full_name` directly from the BDL response (no concatenation, no template). Even a malicious BDL response with HTML in team.name would render as escaped text. Verify: `grep -n "dangerouslySetInnerHTML" vigil-pwa/src/pages/SettingsPage.tsx` matches 0. |
| T-116-05-03 | Denial of Service | Rapid-fire toggling DoS-spamming the PUT endpoint | mitigate | The 400ms debounce coalesces toggle bursts: each toggle resets the timer; only the FINAL state at debounce-fire is PUT. A 100-toggle burst within 400ms produces ONE PUT. Asserted indirectly by Test 2 (single toggle → single PUT after debounce). Server-side rate-limit (existing 100 req/60s) is the second line of defense. |
| T-116-05-04 | Tampering | Optimistic UI desync from server state | mitigate | `lastSavedSportsRef` captures the server-confirmed value on mount AND on each successful PUT. On PUT failure, UI rolls back to lastSavedSportsRef + toast surfaces. On success, lastSavedSportsRef advances to the just-saved value. No "ghost" state can persist past a failed save. Asserted by Test 8. |
| T-116-05-05 | Information Disclosure | Toast leaking sensitive error details | accept | Toast copy is hardcoded to "Couldn't save sports settings — try again." per UI-SPEC and D-21. Server `error.message` is NOT propagated to the toast. League/team picks are not secrets even if leaked. |
| T-116-05-06 | Spoofing | League param in URL (e.g. `/v1/sports/teams/${league}`) is unbounded | mitigate | The `league` argument to `getSportsTeams(league)` is typed `League = 'mlb' | 'nfl' | 'nba' | 'nhl'` at the helper. Caller code only passes literal League values from the LEAGUE_ORDER constant or from server-validated state. Defense in depth: server-side allowlist in Plan 02 rejects any unknown league with 400. |
| T-116-05-07 | Information Disclosure | Sports section visible to logged-out users | mitigate | The entire SettingsPage is gated on `isAuthenticated` in App.tsx (route guard). vigilFetch's existing 'Session expired' redirect (api/client.ts:67-91) handles JWT expiry by bouncing to /auth. No new authentication surface added. |
| T-116-05-08 | Tampering | Mass-assignment via crafted PUT body from a tampered PWA | accept | Server-side validator (Plan 01) rejects extra top-level keys with 400. The PWA helper `setSportsSelections(s)` only sends `{ enabledLeagues, favoriteTeams }` (the function literally JSON.stringifies the typed argument). No client-side path constructs the body with arbitrary keys. |
