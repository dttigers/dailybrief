         name: string
         color: string | null
         primary: boolean
       }

       /** Discriminated union mirroring the GET /v1/calendar/list response shape. */
       export type CalendarListResult =
         | { status: 'ok'; calendars: CalendarInfo[] }
         | { status: 'needs_reauth' }
         | { status: 'error'; error: string }

       /**
        * Fetches the user's Google calendars for the source picker.
        * Returns the full discriminated-union response — caller routes on status.
        * Throws only on transport failures (HTTP non-200 OR network error).
        */
       export async function getCalendarList(): Promise<CalendarListResult> {
         const res = await vigilFetch('/v1/calendar/list')
         if (!res.ok) throw new Error(`Failed to fetch calendar list: ${res.status}`)
         return (await res.json()) as CalendarListResult
       }

       /**
        * Persists the user's calendar selection. Server overwrites wholesale.
        * Empty array IS valid input — see CAL-01 D-11 (empty = all calendars).
        * Cap is enforced server-side at 1000 IDs (T-115-01-02).
        */

         const res = await vigilFetch('/v1/calendar/selections', {
           method: 'PUT',
           body: JSON.stringify({ selectedCalendarIds: ids }),
         })
         if (!res.ok) throw new Error(`Failed to save calendar selections: ${res.status}`)
       }

| T-115-02-01 | Spoofing | Cross-site request forging PUT /v1/calendar/selections | mitigate | The PUT goes through `vigilFetch` which attaches `Authorization: Bearer <jwt>` from sessionStorage (vigil-pwa/src/api/client.ts:53-58). Cross-site requests CANNOT read sessionStorage and CANNOT attach the Authorization header. CSRF is structurally impossible because no cookies are involved. Verify: `grep -n "Authorization.*Bearer" vigil-pwa/src/api/client.ts`. |
| T-115-02-02 | Information Disclosure | Calendar names rendered as raw HTML (XSS) | mitigate | All calendar names are rendered via React text nodes (`{cal.name}`, `{cal.id}`). React escapes by default — no `dangerouslySetInnerHTML`, no `eval`. The color swatch uses `style={{ background: cal.color ?? '#6b7280' }}` — React's style object does NOT execute as CSS expressions; values are coerced to strings and applied via DOM Style API. A malicious `cal.color` containing `;background:url(javascript:...)` would not execute (modern browsers + React style serialization). Verify: `grep -n "dangerouslySetInnerHTML" vigil-pwa/src/pages/SettingsPage.tsx` matches 0 (no occurrences). |
| T-115-02-03 | Denial of Service | Rapid-fire toggling DoS-spamming the PUT endpoint | mitigate | The 400ms debounce coalesces toggle bursts: each toggle resets the timer; only the FINAL state at debounce-fire is PUT. A 100-toggle burst within 400ms produces ONE PUT. Asserted indirectly by Test 2 (single toggle → single PUT after debounce). Server-side rate-limit (existing 100 req/60s) is the second line of defense. |
| T-115-02-04 | Tampering | Optimistic UI desync from server state | mitigate | `lastSavedSelectionRef` captures the server-confirmed value. On PUT failure, UI rolls back to lastSavedSelectionRef + toast surfaces. On success, lastSavedSelectionRef advances to the just-saved value. No "ghost" state can persist past a failed save. |
