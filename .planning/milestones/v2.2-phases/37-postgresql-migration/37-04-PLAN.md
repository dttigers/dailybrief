  ```typescript
  // Total count (excluding soft-deleted)

    count: count(),

    .groupBy(thoughts.category);

  // Task counts by status

    taskStatus: thoughts.taskStatus,
    count: count(),
  }).from(thoughts)
    .where(and(eq(thoughts.category, 'task'), ne(thoughts.syncStatus, 'pendingDeletion')))
    .groupBy(thoughts.taskStatus);

  // Favorites count
  const [{ favCount }] = await db.select({ favCount: count() })
    .from(thoughts)
    .where(and(eq(thoughts.isFavorited, true), ne(thoughts.syncStatus, 'pendingDeletion')));

  // Linked thoughts count
  const [{ linkedCount }] = await db.select({ linkedCount: sql<number>`count(distinct ${thoughtLinks.sourceThoughtId})` })
    .from(thoughtLinks);

  // Recent 5 thoughts
  const recent = await db.select().from(thoughts)
    .where(ne(thoughts.syncStatus, 'pendingDeletion'))
    .orderBy(desc(thoughts.createdAt))
    .limit(5);
  ```

  Assemble response matching the existing shape.

  **brief.ts — GET /brief:**
  Similar aggregation queries plus:

  ```typescript
  // Today's capture count — use PostgreSQL date comparison
  const [{ todayCount }] = await db.select({ todayCount: count() })
    .from(thoughts)
    .where(and(
      sql`${thoughts.createdAt}::date = CURRENT_DATE`,
      ne(thoughts.syncStatus, 'pendingDeletion')
    ));

  // Open/in-progress tasks (limit 10)
  const openTasks = await db.select().from(thoughts)
    .where(and(
      eq(thoughts.category, 'task'),
      or(eq(thoughts.taskStatus, 'open'), eq(thoughts.taskStatus, 'inProgress')),
      ne(thoughts.syncStatus, 'pendingDeletion')
    ))
    .orderBy(desc(thoughts.createdAt))
    .limit(10);

  // Recent therapy thoughts
  const therapyThoughts = await db.select().from(thoughts)
    .where(and(
      eq(thoughts.category, 'therapy'),
      ne(thoughts.syncStatus, 'pendingDeletion')
    ))
    .orderBy(desc(thoughts.createdAt))
    .limit(5);

  // Unprocessed count (no category)
  const [{ unprocessed }] = await db.select({ unprocessed: count() })
    .from(thoughts)
    .where(and(
      sql`${thoughts.category} IS NULL`,
      ne(thoughts.syncStatus, 'pendingDeletion')
    ));
  ```
