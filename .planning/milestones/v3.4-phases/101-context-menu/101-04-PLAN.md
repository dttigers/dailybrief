```tsx
useEffect(() => {
  const visibleItemCount = /* computed from current view + props.projects */
  const onKey = (e: KeyboardEvent) => {
    if (e.key === 'Escape') { e.preventDefault(); onClose(); return }
    if (e.key === 'ArrowDown') {
      e.preventDefault()
      setFocusedIndex((i) => (i + 1) % visibleItemCount)
    }
    if (e.key === 'ArrowUp') {
      e.preventDefault()
      setFocusedIndex((i) => (i - 1 + visibleItemCount) % visibleItemCount)
    }
    if (e.key === 'Enter' || e.key === ' ') {
      e.preventDefault()
      itemRefs.current[focusedIndex]?.click()
    }
    if (e.key === 'ArrowRight' && view === 'root') {
      // Open submenu if focused item is Move-to-category or Add-to-project
      if (focusedIndex === 2 && openedVia === 'touch') setView('categories')
      if (focusedIndex === 2 && openedVia === 'mouse') setDesktopSubmenu('categories')
      if (focusedIndex === 3 && openedVia === 'touch') setView('projects')
      if (focusedIndex === 3 && openedVia === 'mouse') setDesktopSubmenu('projects')
    }
    if (e.key === 'ArrowLeft') {
      if (view === 'categories' || view === 'projects') setView('root')
      if (desktopSubmenu !== null) setDesktopSubmenu(null)
    }
  }
  window.addEventListener('keydown', onKey)
  return () => window.removeEventListener('keydown', onKey)
}, [onClose, view, focusedIndex, openedVia, /* projects.length */])
```
