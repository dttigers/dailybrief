    ```swift
    import Foundation

    /// Per-session timer state machine. One actor per active sessionId.
    /// Composition: WatcherActor (Plan 08) routes ParsedLine to the matching SessionState
    /// via process(line:); a global 1Hz Task (Plan 09) calls evaluate(now:config:) once per second
    /// per active session.
    public actor SessionState {

        public let sessionId: String

        // Most recent line timestamp (advanced on every process()).
        public private(set) var latestLineTimestamp: Date = Date.distantPast
        // Most recent assistant.message.stop_reason.
        public private(set) var latestStopReason: String? = nil
        // Most recent user line's permissionMode (drives needs_input gating).
        public private(set) var latestPermissionMode: String? = nil
        // sessionHadError NEVER resets within session (Pitfall 3 mitigation).
        public private(set) var sessionHadError: Bool = false
        // Pending tool_use IDs awaiting tool_result (assistant emitted, no matching user line yet).
        // Map: id → (assistant line timestamp, byte offset)
        public private(set) var pendingToolUses: [String: (Date, UInt64)] = [:]
        // Most recent emission timestamps (drive debounce / dedupe / suppression).
        public private(set) var lastNeedsInputEmittedAt: Date = Date.distantPast
        public private(set) var lastHeartbeatEmittedAt: Date = Date.distantPast
        public private(set) var taskCompleteEmittedAt: Date? = nil
        public private(set) var taskFailedEmittedAt: Date? = nil
        // Latest assistant line offset — used for synthesized event hashing (heartbeat / task_complete fire
        // without their own JSONL line; we hash against the latest line's offset for restart-replay safety).
        public private(set) var latestLineOffset: UInt64 = 0
        // Label fields propagated to VigilPayload.
        public private(set) var label: String = ""
        public private(set) var host: String = ""

        public init(sessionId: String, label: String, host: String) {
            self.sessionId = sessionId
            self.label = label
            self.host = host
        }

        // MARK: - process(line:) — update state from each parsed line, emit any IMMEDIATE-fire events

        /// Returns events that fire immediately at line-arrival time (currently: task_failed only).
        /// Returns [] for events that require timer evaluation (needs_input, task_complete, heartbeat,
        /// milestone — milestone is handled separately by Plan 05's MilestoneMatcher in Plan 09's wiring).
        public func process(line: ParsedLine) -> [VigilPayload] {
            // Skip non-spec lines defensively (caller should already filter, but be safe).
            guard line.lineType != .nonSpec else { return [] }

            // Update common timer state.
            if line.timestamp > latestLineTimestamp {
                latestLineTimestamp = line.timestamp
                latestLineOffset = line.byteOffset
            }
            // After a new line arrives, the heartbeat suppression window resets:
            // heartbeat fires only after sustained silence, so reset emission stamp so a future silence triggers.
            // (We DO NOT reset lastHeartbeatEmittedAt directly — instead, evaluate() compares to latestLineTimestamp.)

            switch line.lineType {

            case .user:
                // Track permission mode for needs_input gating.
                if let pm = line.permissionMode { latestPermissionMode = pm }
                // Resolve any pending tool_uses whose tool_use_id matches a tool_result on this line.
                // (toolResultErrors carries IDs whose is_error==true; successful tool_results are NOT in toolResultErrors,
                // but we still need to remove them from pendingToolUses. Plan 02 ParsedLine carries toolResultErrors only —
                // for needs_input gap resolution we treat ANY tool_result as resolving. We use a simpler heuristic:
                // when ANY user line arrives, all pending tool_uses older than the line are presumed resolved.)
                // PRAGMATIC RULE: clear pendingToolUses on every user line arrival. Phase 120 README confirms a single
                // tool_use is resolved by the next user line carrying tool_result with matching tool_use_id; pragmatic
                // approximation here is "any user line clears the gap" — false positives cost negligible.
                pendingToolUses.removeAll()

                // task_failed: immediate fire on is_error:true (single-field discriminator per Phase 120 D-04 README).
                // Deduped: emit once per session; subsequent errors suppressed until session changes.
                if !line.toolResultErrors.isEmpty && taskFailedEmittedAt == nil {
                    sessionHadError = true
                    taskFailedEmittedAt = line.timestamp
                    let payload = makePayload(event: .taskFailed,
                                              message: "tool_result is_error=true (\(line.toolResultErrors.count) failed tool_use_id(s))",
                                              timestamp: line.timestamp,
                                              byteOffset: line.byteOffset,
                                              exitCode: nil)
                    return [payload]
                } else if !line.toolResultErrors.isEmpty {
                    sessionHadError = true   // mark even when suppressing duplicate emission
                }

            case .assistant:
                // Track stop_reason for task_complete heuristic.
                latestStopReason = line.stopReason
                // Add any new tool_use IDs to pending.
                for tid in line.toolUseIds {
                    pendingToolUses[tid] = (line.timestamp, line.byteOffset)
                }

            case .nonSpec: break
            }

            return []
        }

        // MARK: - evaluate(now:config:) — timer-driven events fired by the 1Hz tick loop in Plan 09

        public func evaluate(now: Date = Date(), config: WatchConfig) -> [VigilPayload] {
            var out: [VigilPayload] = []

            // 1) needs_input: any pending tool_use older than gap_seconds AND permissionMode != bypassPermissions.
            //    Debounced: at most once per debounce_seconds.
            if (latestPermissionMode ?? "default") != "bypassPermissions" {
                let gap = TimeInterval(config.needsInputGapSeconds)
                let debounce = TimeInterval(config.needsInputDebounceSeconds)
                let oldestPending = pendingToolUses.values.map { $0.0 }.min()
                if let oldest = oldestPending,
                   now.timeIntervalSince(oldest) >= gap,
                   now.timeIntervalSince(lastNeedsInputEmittedAt) >= debounce {
                    let pendingId = pendingToolUses.first?.key ?? "unknown"
                    let pendingOffset = pendingToolUses.first?.value.1 ?? latestLineOffset
                    out.append(makePayload(event: .needsInput,
                                            message: "tool_use \(pendingId) awaiting approval (\(Int(now.timeIntervalSince(oldest)))s)",
                                            timestamp: now,
                                            byteOffset: pendingOffset,
                                            exitCode: nil))
                    lastNeedsInputEmittedAt = now
                }
            }

            // 2) task_complete OR task_failed-by-precedence.
            //    Trigger: latestStopReason == "end_turn" AND silence > task_complete_silence_seconds.
            //    If sessionHadError, prefer task_failed (re-emit if not yet emitted, otherwise suppress).
            let silenceTC = TimeInterval(config.taskCompleteSilenceSeconds)
            if latestStopReason == "end_turn",
               now.timeIntervalSince(latestLineTimestamp) >= silenceTC,
               taskCompleteEmittedAt == nil {
                if sessionHadError {
                    // Precedence: re-emit task_failed only if not already emitted (deduped).
                    if taskFailedEmittedAt == nil {
                        taskFailedEmittedAt = now
                        out.append(makePayload(event: .taskFailed,
                                                message: "task ended with errors (precedence over task_complete)",
                                                timestamp: now,
                                                byteOffset: latestLineOffset,
                                                exitCode: nil))
                    }
                    // Mark task_complete-emitted to prevent re-evaluation, even though we suppressed it.
                    taskCompleteEmittedAt = now
                } else {
                    taskCompleteEmittedAt = now
                    out.append(makePayload(event: .taskComplete,
                                            message: "stop_reason=end_turn + \(Int(silenceTC))s silence",
                                            timestamp: now,
                                            byteOffset: latestLineOffset,
                                            exitCode: nil))
                }
            }

            // 3) heartbeat: silence > heartbeat_seconds AND we haven't already emitted a heartbeat
            //    for this silence window. Reset window: a new line arriving advances latestLineTimestamp,
            //    making the next heartbeat eligible.
            let silenceHB = TimeInterval(config.heartbeatSeconds)
            if now.timeIntervalSince(latestLineTimestamp) >= silenceHB,
               lastHeartbeatEmittedAt < latestLineTimestamp || lastHeartbeatEmittedAt == Date.distantPast {
                lastHeartbeatEmittedAt = now
                out.append(makePayload(event: .heartbeat,
                                        message: "session idle > \(Int(silenceHB))s",
                                        timestamp: now,
                                        byteOffset: latestLineOffset,
                                        exitCode: nil))
            }

            return out
        }

        // MARK: - VigilPayload assembly

        private func makePayload(event: VigilEvent, message: String?, timestamp: Date,
                                 byteOffset: UInt64, exitCode: Int?) -> VigilPayload {
            let f = ISO8601DateFormatter()
            f.formatOptions = [.withInternetDateTime, .withFractionalSeconds]
            return VigilPayload(
                sessionId: sessionId,
                event: event.rawValue,
                message: message,
                timestamp: f.string(from: timestamp),
                label: label,
                host: host,
                exitCode: exitCode,
                clientEventId: makeClientEventId(sessionId: sessionId,
                                                 byteOffset: byteOffset,
                                                 eventType: event.rawValue)
            )
        }

        // MARK: - Test inspection helpers

        public func setLatestLineTimestampForTest(_ d: Date) { latestLineTimestamp = d }
        public func setLastHeartbeatForTest(_ d: Date) { lastHeartbeatEmittedAt = d }
    }
    ```

    Implementation notes:
    - The pragmatic "clear pending tool_uses on any user line" rule is documented in the action — it trades a small false-negative window (legit chained tool calls within the same user line) for robust gap-resolution. Phase 122 RESEARCH.md flagged this as acceptable for the narrow scope.
    - `sessionHadError` flag is `var private(set)` — visible to tests, mutated only via process().
    - `evaluate()` is idempotent in the sense that re-calling it without new lines arriving produces no events (because the dedupe stamps catch each event class).
    - Heartbeat window reset is detected via `lastHeartbeatEmittedAt < latestLineTimestamp` — when a new line lands, latestLineTimestamp jumps forward, eligibility re-arms.
    - All synthesized timer events (`needs_input`, `task_complete`, `heartbeat`) hash against the LATEST or PENDING-tool-use line offset — this is sufficient for D-01 restart-replay safety because re-running the daemon over the same JSONL produces the same offsets.

    ```swift
    final class SessionActorTests: XCTestCase {

        private let cfg: WatchConfig = .defaults
        private let baseTime = ISO8601DateFormatter().date(from: "2026-05-08T18:00:00Z")!

        private func makeAssistantToolUse(sessionId: String, atOffset: UInt64, t: Date,
                                          toolUseId: String) -> ParsedLine {
            ParsedLine(sessionId: sessionId, lineType: .assistant, timestamp: t, byteOffset: atOffset,
                       cwd: nil, permissionMode: nil, stopReason: "tool_use",
                       toolUseIds: [toolUseId], toolResultErrors: [], assistantTextBlocks: [])
        }

        private func makeAssistantEndTurn(sessionId: String, atOffset: UInt64, t: Date) -> ParsedLine {
            ParsedLine(sessionId: sessionId, lineType: .assistant, timestamp: t, byteOffset: atOffset,
                       cwd: nil, permissionMode: nil, stopReason: "end_turn",
                       toolUseIds: [], toolResultErrors: [], assistantTextBlocks: [])
        }

        private func makeUserToolResult(sessionId: String, atOffset: UInt64, t: Date,
                                        permission: String = "default", errors: [String] = []) -> ParsedLine {
            ParsedLine(sessionId: sessionId, lineType: .user, timestamp: t, byteOffset: atOffset,
                       cwd: nil, permissionMode: permission, stopReason: nil,
                       toolUseIds: [], toolResultErrors: errors, assistantTextBlocks: [])
        }

        // MARK: - needs_input (D-03 + D-05 debounce)

        func testNeedsInputFiresAfter10sGap() async {
            let s = SessionState(sessionId: "s1", label: "L", host: "H")
            // First a user line to set permissionMode
            await s.process(line: makeUserToolResult(sessionId: "s1", atOffset: 0, t: baseTime))
            // Assistant emits tool_use at t+0
            await s.process(line: makeAssistantToolUse(sessionId: "s1", atOffset: 100, t: baseTime, toolUseId: "tu_1"))
            // Evaluate at t+5 — should NOT fire (gap < 10)
            let early = await s.evaluate(now: baseTime.addingTimeInterval(5), config: cfg)
            XCTAssertTrue(early.isEmpty)
            // Evaluate at t+11 — should fire
            let late = await s.evaluate(now: baseTime.addingTimeInterval(11), config: cfg)
            XCTAssertEqual(late.count, 1)
            XCTAssertEqual(late[0].event, "needs_input")
        }

        func testNeedsInputSuppressedByBypassPermissions() async {
            let s = SessionState(sessionId: "s1", label: "L", host: "H")
            await s.process(line: makeUserToolResult(sessionId: "s1", atOffset: 0, t: baseTime, permission: "bypassPermissions"))
            await s.process(line: makeAssistantToolUse(sessionId: "s1", atOffset: 100, t: baseTime, toolUseId: "tu_1"))
            let result = await s.evaluate(now: baseTime.addingTimeInterval(60), config: cfg)
            // No needs_input event — bypass mode suppresses it.
            XCTAssertFalse(result.contains { $0.event == "needs_input" })
        }

        func testNeedsInputDebouncedTo30sWindow() async {
            let s = SessionState(sessionId: "s1", label: "L", host: "H")
            await s.process(line: makeUserToolResult(sessionId: "s1", atOffset: 0, t: baseTime))
            await s.process(line: makeAssistantToolUse(sessionId: "s1", atOffset: 100, t: baseTime, toolUseId: "tu_1"))
    ```
