    import Foundation

    /// Stdout NDJSON shape per CONTEXT.md "Logging surface".
    public struct EventLogLine: Encodable, Sendable {
        public let ts: String                  // ISO-8601 — wall clock at emit attempt time
        public let sessionId: String
        public let event: String
        public let message: String?
        public let label: String
        public let host: String
        public let exitCode: Int?
        public let clientEventId: String
        public let postStatus: Int             // HTTP status code, or 0 if queued (not posted yet)

        public enum CodingKeys: String, CodingKey {
            case ts
            case sessionId = "session_id"
            case event, message, label, host
            case exitCode = "exit_code"
            case clientEventId = "client_event_id"
            case postStatus = "post_status"
        }

        public init(ts: String, sessionId: String, event: String, message: String?,
                    label: String, host: String, exitCode: Int?, clientEventId: String, postStatus: Int) {
            self.ts = ts; self.sessionId = sessionId; self.event = event
            self.message = message; self.label = label; self.host = host
            self.exitCode = exitCode; self.clientEventId = clientEventId
            self.postStatus = postStatus
        }
    }

    public func logEvent(_ line: EventLogLine) {
        guard let data = try? JSONEncoder().encode(line),
              let s = String(data: data, encoding: .utf8) else { return }
        print(s)        // print() uses stdout; Plan 09 main.swift sets setbuf(stdout, nil)
    }

    /// Mask any Bearer token substring to prevent T-122-01 (bearer leak in stderr).
    /// Matches `Bearer <non-space>+` (case-insensitive) and replaces the token with `Bearer <redacted>`.
    public func maskBearer(_ s: String) -> String {
        guard let regex = try? NSRegularExpression(pattern: #"(?i)Bearer\s+\S+"#) else { return s }
        let range = NSRange(s.startIndex..., in: s)
        return regex.stringByReplacingMatches(in: s, range: range, withTemplate: "Bearer <redacted>")
    }

    nonisolated(unsafe) private let stderrHandle = FileHandle.standardError

    public func logInfo(_ msg: String)  { stderrHandle.write(Data("[INFO]  \(maskBearer(msg))\n".utf8)) }
    public func logWarn(_ msg: String)  { stderrHandle.write(Data("[WARN]  \(maskBearer(msg))\n".utf8)) }
    public func logError(_ msg: String) { stderrHandle.write(Data("[ERROR] \(maskBearer(msg))\n".utf8)) }

    /// ISO-8601 helper — emits "2026-05-08T18:08:24.330Z" shape.
    public func nowISO8601() -> String {
        let f = ISO8601DateFormatter()
        f.formatOptions = [.withInternetDateTime, .withFractionalSeconds]
        return f.string(from: Date())
    }
    ```

    import Foundation

    /// Abstraction over URLSession.data(for:) so EmitterTests can inject deterministic responses.
    public protocol HTTPClient: Sendable {
        func data(for request: URLRequest) async throws -> (Data, URLResponse)
    }

    /// Production adapter: real URLSession with .ephemeral configuration.
    public struct DefaultHTTPClient: HTTPClient {
        private let session: URLSession
        public init() {
            self.session = URLSession(configuration: .ephemeral)
        }
        public func data(for request: URLRequest) async throws -> (Data, URLResponse) {
            try await session.data(for: request)
        }
    }

    /// Single global emitter. Owns the in-memory queue + retry/backoff state.
    /// Sessions and watcher push VigilPayload via `enqueue(_:)`. A flush loop drains the queue.
    public actor EmitterActor {

        public static let queueCapacity = 100  // CONTEXT.md "100-event in-memory queue"
        public static let maxAttempts = 6      // CONTEXT.md "Max 6 attempts before queueing for later"

        private var config: WatchConfig
        private let http: HTTPClient
        private var queue: [VigilPayload] = []
        private var quarantined: Bool = false
        private var flushTask: Task<Void, Never>? = nil
        private var sleepFn: (Duration) async throws -> Void

        public init(config: WatchConfig, http: HTTPClient = DefaultHTTPClient(),
                    sleepFn: @escaping (Duration) async throws -> Void = { try await Task.sleep(for: $0) }) {
            self.config = config
            self.http = http
            self.sleepFn = sleepFn
            self.quarantined = config.apiKey.isEmpty
        }

        // MARK: - Public surface

        public func enqueue(_ payload: VigilPayload) {
            // FIFO with cap — drop oldest on overflow per CONTEXT.md.
            if queue.count >= Self.queueCapacity {
                let dropped = queue.removeFirst()
                logWarn("queue overflow — dropped oldest event client_event_id=\(dropped.clientEventId)")
            }
            queue.append(payload)
            // NDJSON log: 0 = "queued, not yet posted"
            logEvent(EventLogLine(
                ts: nowISO8601(), sessionId: payload.sessionId, event: payload.event,
                message: payload.message, label: payload.label, host: payload.host,
                exitCode: payload.exitCode, clientEventId: payload.clientEventId, postStatus: 0
            ))
        }

        /// Start a background Task that loops: pull head of queue, attempt POST, retry/drop per rules.
        public func startFlushLoop() {
            guard flushTask == nil else { return }
            flushTask = Task { [weak self] in
                while !Task.isCancelled {
                    guard let self else { return }
                    await self.flushOnce()
                    try? await self.sleepFn(.seconds(1))
                }
            }
        }

        public func stopFlushLoop() {
            flushTask?.cancel()
            flushTask = nil
        }

        /// CONTEXT.md "5s SIGTERM drain" — best-effort flush all queued events with wall-clock budget.
        /// Returns: number of events successfully flushed (or treated as drop) before deadline.
        public func drain(deadlineSeconds: TimeInterval = 5.0) async -> Int {
            let start = Date()
            var processed = 0
            while !queue.isEmpty {
                let elapsed = Date().timeIntervalSince(start)
                if elapsed >= deadlineSeconds {
                    logWarn("drain deadline reached — \(queue.count) events lost")
                    return processed
                }
                let payload = queue.removeFirst()
                _ = try? await postOnce(payload)   // single attempt, no retry sleep during drain
                processed += 1
            }
            logInfo("drain complete — \(processed) events flushed before deadline")
            return processed
        }

        // MARK: - Inspection (for tests)

        public func queueDepth() -> Int { queue.count }
        public func isQuarantined() -> Bool { quarantined }

        // MARK: - Private flush + post

        /// Pull one event off the queue, attempt up to maxAttempts with backoff, drop or requeue on permanent failure.
        private func flushOnce() async {
            guard !quarantined else {
                if !queue.isEmpty {
                    logWarn("quarantine: api_key missing — \(queue.count) events buffered, not posting")
                }
                return
            }
            guard !queue.isEmpty else { return }
            let payload = queue.removeFirst()
            for attempt in 1...Self.maxAttempts {
                let outcome = await postWithBackoff(payload: payload, attempt: attempt)
                switch outcome {
                case .success:        return    // committed
                case .drop4xx:        return    // permanent client error — log + drop
                case .retry(let d):
                    try? await sleepFn(d)
                    continue
                }
            }
            // Exhausted attempts — requeue at the BACK of the queue per CONTEXT.md
            // ("the event sits in the in-memory queue until the next emit triggers a fresh attempt sweep").
            queue.append(payload)
        }

        private enum Outcome { case success; case drop4xx; case retry(Duration) }

        private func postWithBackoff(payload: VigilPayload, attempt: Int) async -> Outcome {
            do {
                let (status, retryAfter) = try await postOnce(payload)
                switch status {
                case 200, 201:                return .success      // 200 = idempotent dup (D-C2), 201 = inserted
                case 429:                     return .retry(retryAfter ?? backoffDelay(attempt: attempt))
                case 400...499:
                    logWarn("permanent 4xx \(status) — dropping client_event_id=\(payload.clientEventId)")
                    return .drop4xx
                default:                      return .retry(backoffDelay(attempt: attempt))     // 5xx etc
                }
            } catch {
                // Transient URLError or unexpected — retry with backoff.
                logWarn("post attempt \(attempt) failed: \(error.localizedDescription)")
                return .retry(backoffDelay(attempt: attempt))
            }
        }

        /// Single POST, returns (statusCode, parsedRetryAfter). No retry inside.
        @discardableResult
        public func postOnce(_ payload: VigilPayload) async throws -> (status: Int, retryAfter: Duration?) {
            guard let url = URL(string: "\(config.apiURL)/v1/agent-events") else {
                throw URLError(.badURL)
            }
            var req = URLRequest(url: url)
            req.httpMethod = "POST"
            req.setValue("Bearer \(config.apiKey)", forHTTPHeaderField: "Authorization")
            req.setValue("application/json", forHTTPHeaderField: "Content-Type")
            req.httpBody = try JSONEncoder().encode(payload)
            req.timeoutInterval = 10

            let (_, response) = try await http.data(for: req)
            guard let http = response as? HTTPURLResponse else {
                throw URLError(.badServerResponse)
            }
            let retryAfter = parseRetryAfter(http)
            // NDJSON status update line — re-emit with actual post_status for tail consumers.
            logEvent(EventLogLine(
                ts: nowISO8601(), sessionId: payload.sessionId, event: payload.event,
                message: payload.message, label: payload.label, host: payload.host,
                exitCode: payload.exitCode, clientEventId: payload.clientEventId, postStatus: http.statusCode
            ))
            return (http.statusCode, retryAfter)
        }

        // MARK: - Backoff math (testable)

        /// Exponential backoff with ±25% jitter. Verified against CONTEXT.md sequence 1/2/4/8/16/32s cap.
        public static func backoffDelay(attempt: Int, jitterFn: () -> Double = { Double.random(in: 0.75...1.25) }) -> Duration {
            let base = pow(2.0, Double(attempt - 1))   // 1, 2, 4, 8, 16, 32, 64...
            let capped = min(base, 32.0)
            let jittered = capped * jitterFn()
            return .seconds(jittered)
        }

        /// Instance-side wrapper around the static. Allows actor-internal callers to use without specifying jitter.
        private func backoffDelay(attempt: Int) -> Duration {
            return Self.backoffDelay(attempt: attempt)
        }

        /// HTTP/1.1 RFC 9110 §10.2.4 — Retry-After is either delta-seconds OR HTTP-date.
        public static func parseRetryAfter(_ response: HTTPURLResponse) -> Duration? {
            guard let value = response.value(forHTTPHeaderField: "Retry-After") else { return nil }
            if let seconds = Double(value) { return .seconds(max(0, seconds)) }
            let f = DateFormatter()
            f.locale = Locale(identifier: "en_US_POSIX")
            f.dateFormat = "EEE, dd MMM yyyy HH:mm:ss zzz"
            if let date = f.date(from: value) {
                let delta = date.timeIntervalSinceNow
                return .seconds(max(1, delta))
            }
            return nil
        }
        private func parseRetryAfter(_ r: HTTPURLResponse) -> Duration? { Self.parseRetryAfter(r) }
    }
    ```
