    import Foundation

    public enum TOMLValue: Sendable {
        case string(String)
        case integer(Int)
        case stringArray([String])
    }

    public enum TOMLParseError: Error {
        case unterminatedArray
        case malformedKey(String)
        case unsupportedValue(String)
    }

    public struct TOMLParser {
        public init() {}

        /// Parses a TOML document into a flat dictionary. Supports:
        /// - `key = "string"` (single or double quotes)
        /// - `key = 123` (integer)
        /// - `key = ["s1", "s2", ...]` multi-line OR single-line array of strings
        /// - `# comment` line and trailing inline `# comment` (only when not inside a string)
        /// Does NOT support: nested tables `[section]`, inline tables `{}`, booleans, floats, dates.
        public func parse(_ source: String) throws -> [String: TOMLValue] {
            var result: [String: TOMLValue] = [:]
            var lines = source.components(separatedBy: "\n")
            var i = 0
            while i < lines.count {
                let raw = lines[i]
                let stripped = stripCommentAndTrim(raw)
                if stripped.isEmpty { i += 1; continue }
                guard let eq = stripped.firstIndex(of: "=") else { i += 1; continue }
                let key = String(stripped[..<eq]).trimmingCharacters(in: .whitespaces)
                var rest = String(stripped[stripped.index(after: eq)...]).trimmingCharacters(in: .whitespaces)
                if key.isEmpty { throw TOMLParseError.malformedKey(raw) }

                if rest.hasPrefix("\"") || rest.hasPrefix("'") {
                    // String value: strip surrounding quote
                    result[key] = .string(unquote(rest))
                } else if let intValue = Int(rest) {
                    result[key] = .integer(intValue)
                } else if rest.hasPrefix("[") {
                    // Array — may continue across multiple lines until closing ']'
                    var arrayBuf = rest
                    while !arrayBuf.contains("]") {
                        i += 1
                        guard i < lines.count else { throw TOMLParseError.unterminatedArray }
                        arrayBuf += " " + stripCommentAndTrim(lines[i])
                    }
                    result[key] = .stringArray(parseStringArray(arrayBuf))
                } else {
                    throw TOMLParseError.unsupportedValue("key=\(key) value=\(rest)")
                }
                i += 1
            }
            return result
        }

        private func stripCommentAndTrim(_ line: String) -> String {
            // Remove `# ...` UNLESS the # appears inside a string literal.
            var inString = false
            var out = ""
            for c in line {
                if c == "\"" { inString.toggle() }
                if c == "#" && !inString { break }
                out.append(c)
            }
            return out.trimmingCharacters(in: .whitespaces)
        }

        private func unquote(_ s: String) -> String {
            var t = s.trimmingCharacters(in: .whitespaces)
            if (t.hasPrefix("\"") && t.hasSuffix("\"")) || (t.hasPrefix("'") && t.hasSuffix("'")) {
                t = String(t.dropFirst().dropLast())
            }
            // Decode \\ \n \" escapes (minimal subset; expand if a regex pattern needs it)
            t = t.replacingOccurrences(of: "\\\\", with: "\\")
            t = t.replacingOccurrences(of: "\\\"", with: "\"")
            return t
        }

        private func parseStringArray(_ buf: String) -> [String] {
            // Strip surrounding [ ... ]
            guard let lb = buf.firstIndex(of: "["), let rb = buf.lastIndex(of: "]") else { return [] }
            let inner = String(buf[buf.index(after: lb)..<rb])
            // Split on commas that live OUTSIDE quoted strings
            var elements: [String] = []
            var current = ""
            var inString = false
            var quoteChar: Character = "\""
            for c in inner {
                if !inString && (c == "\"" || c == "'") { inString = true; quoteChar = c; current.append(c); continue }
                if inString && c == quoteChar { inString = false; current.append(c); continue }
                if !inString && c == "," {
                    let trimmed = current.trimmingCharacters(in: .whitespaces)
                    if !trimmed.isEmpty { elements.append(unquote(trimmed)) }
                    current = ""
                    continue
                }
                current.append(c)
            }
            let last = current.trimmingCharacters(in: .whitespaces)
            if !last.isEmpty { elements.append(unquote(last)) }
            return elements
        }
    }

    public struct WatchConfig: Sendable {
        public var apiURL: String
        public var apiKey: String                       // empty string → quarantine state
        public var heartbeatSeconds: Int                // D-05 locked default 60
        public var needsInputDebounceSeconds: Int       // D-05 locked default 30
        public var needsInputGapSeconds: Int            // D-03 default 10
        public var taskCompleteSilenceSeconds: Int      // D-04 default 30
        public var projectsDir: String                  // ~ expansion done at use site

        public var milestonePatterns: [String]          // 6-pattern starter per D-10

        public static let defaults = WatchConfig(
            apiURL: "https://api.vigilhub.io",
            apiKey: "",
            heartbeatSeconds: 60,
            needsInputDebounceSeconds: 30,
            needsInputGapSeconds: 10,
            taskCompleteSilenceSeconds: 30,
            projectsDir: "~/.claude/projects",
            hostLabel: "",
            milestonePatterns: defaultMilestonePatterns
        )

        public static let defaultMilestonePatterns: [String] = [
            "^[✓✔] ",
            "\\b(all|every)\\s+(tests?|checks?)\\s+(pass|passed|passing)\\b",
            "\\bbuild\\s+(succeeded|successful|complete)\\b",
            "\\b(plan|phase)\\s+\\d+(\\.\\d+)?(-\\d+)?\\s+complete\\b",
            "\\bdeployed\\s+(to\\s+\\S+\\s+)?successfully\\b",
            "\\bPR\\s+#\\d+\\s+(created|merged)\\b",
        ]

        public init(apiURL: String, apiKey: String, heartbeatSeconds: Int,
                    needsInputDebounceSeconds: Int, needsInputGapSeconds: Int,
                    taskCompleteSilenceSeconds: Int, projectsDir: String,
                    hostLabel: String, milestonePatterns: [String]) {
            self.apiURL = apiURL; self.apiKey = apiKey
            self.heartbeatSeconds = heartbeatSeconds
            self.needsInputDebounceSeconds = needsInputDebounceSeconds
            self.needsInputGapSeconds = needsInputGapSeconds
            self.taskCompleteSilenceSeconds = taskCompleteSilenceSeconds
            self.projectsDir = projectsDir; self.hostLabel = hostLabel
            self.milestonePatterns = milestonePatterns
        }
    }

    public enum ConfigError: Error {
        case cannotCreateDirectory(String)
        case cannotWriteDefaults(String)
        case readFailed(String)
        case parseFailed(Error)
    }

    public struct ConfigLoader {

        /// Default config path: `~/.config/vigil/watch.toml` (per AGENT-WATCH-06).
        public static let defaultConfigPath: String = {
            let home = NSHomeDirectory()
            return "\(home)/.config/vigil/watch.toml"
        }()

        /// Default first-run TOML body (commented inline per CONTEXT.md "First-run watch.toml creation").
        public static let defaultTOMLBody: String = """
        # vigil-watch configuration — created on first run
        # Edit and restart the daemon for changes to take effect.

        # Vigil Core API base URL. Override to http://127.0.0.1:3001 for local dev.
        api_url = "https://api.vigilhub.io"

        # Bearer token for vigil-core. If blank, falls back to VIGIL_API_KEY env var.
        # If both are missing, daemon runs in quarantine state — events queue but do not POST.
        api_key = ""

        # D-05 locked: silence threshold for heartbeat emission.
        heartbeat_seconds = 60

        # D-05 locked: one needs_input per session per debounce window.
        needs_input_debounce_seconds = 30

        # D-03: gap between assistant tool_use and matching user tool_result before firing needs_input.
        needs_input_gap_seconds = 10

        # D-04: silence after assistant end_turn before firing task_complete.
        task_complete_silence_seconds = 30

        # Path to ~/.claude/projects/ (parent — daemon enumerates ALL subdirectories).
        # Override to a single namespace subdir to allowlist.
        projects_dir = "~/.claude/projects"

        # Label included on every event payload. Blank → use ProcessInfo.processInfo.hostName.
        host_label = ""

        # D-10: 6-pattern starter, case-insensitive (regex receives implicit (?i) prepend at compile time).
        milestone_patterns = [
            "^[✓✔] ",
            "\\\\b(all|every)\\\\s+(tests?|checks?)\\\\s+(pass|passed|passing)\\\\b",
            "\\\\bbuild\\\\s+(succeeded|successful|complete)\\\\b",
            "\\\\b(plan|phase)\\\\s+\\\\d+(\\\\.\\\\d+)?(-\\\\d+)?\\\\s+complete\\\\b",
            "\\\\bdeployed\\\\s+(to\\\\s+\\\\S+\\\\s+)?successfully\\\\b",
            "\\\\bPR\\\\s+#\\\\d+\\\\s+(created|merged)\\\\b",
        ]
        """

        /// Reads watch.toml. If absent, creates parent directory + writes defaults, then re-reads.
        /// On read OR parse OR mkdir failure: throws — caller should log to stderr and exit non-zero
        /// (per CONTEXT.md "Failure to create the file is a fatal startup error").
        ///
        /// After parsing the file: applies api_key bootstrap precedence:
        /// 1. watch.toml api_key (non-empty) wins
        /// 2. VIGIL_API_KEY env var falls in
        /// 3. Both empty → returns config with apiKey == "" (caller routes to quarantine)
        public static func load(path: String = defaultConfigPath,
                                env: [String: String] = ProcessInfo.processInfo.environment) throws -> WatchConfig {
            let url = URL(fileURLWithPath: path)
            let dir = url.deletingLastPathComponent()
            let fm = FileManager.default

            // Create parent dir if needed.
            if !fm.fileExists(atPath: dir.path) {
                do {
                    try fm.createDirectory(at: dir, withIntermediateDirectories: true)
                } catch {
                    throw ConfigError.cannotCreateDirectory(dir.path + ": \(error)")
                }
            }
            // Create defaults file if absent.
            if !fm.fileExists(atPath: url.path) {
                do {
                    try defaultTOMLBody.write(to: url, atomically: true, encoding: .utf8)
                } catch {
                    throw ConfigError.cannotWriteDefaults(url.path + ": \(error)")
                }
            }
            // Read.
            let source: String
            do { source = try String(contentsOf: url, encoding: .utf8) }
            catch { throw ConfigError.readFailed(url.path + ": \(error)") }

            // Parse.
            let parsed: [String: TOMLValue]
            do { parsed = try TOMLParser().parse(source) }
            catch { throw ConfigError.parseFailed(error) }

            // Map into WatchConfig with defaults for any absent keys.
            var cfg = WatchConfig.defaults
            if case let .string(s) = parsed["api_url"]   { cfg.apiURL = s }
            if case let .string(s) = parsed["api_key"]   { cfg.apiKey = s }
            if case let .integer(i) = parsed["heartbeat_seconds"] { cfg.heartbeatSeconds = i }
            if case let .integer(i) = parsed["needs_input_debounce_seconds"] { cfg.needsInputDebounceSeconds = i }
            if case let .integer(i) = parsed["needs_input_gap_seconds"] { cfg.needsInputGapSeconds = i }
            if case let .integer(i) = parsed["task_complete_silence_seconds"] { cfg.taskCompleteSilenceSeconds = i }
            if case let .string(s) = parsed["projects_dir"] { cfg.projectsDir = s }
            if case let .string(s) = parsed["host_label"]   { cfg.hostLabel = s }
            if case let .stringArray(a) = parsed["milestone_patterns"] { cfg.milestonePatterns = a }

            // api_key precedence: file → VIGIL_API_KEY env → quarantine.
            if cfg.apiKey.isEmpty, let envKey = env["VIGIL_API_KEY"], !envKey.isEmpty {
                cfg.apiKey = envKey
            }

            return cfg
        }

        /// Expand ~ in a path (used for projects_dir resolution).
        public static func expandHome(_ path: String) -> String {
            if path.hasPrefix("~/") {
                return NSHomeDirectory() + String(path.dropFirst(1))
            }
            return path
        }

        /// Resolve effective host label: hostLabel from config wins; blank → ProcessInfo.processInfo.hostName.
        public static func resolveHost(_ cfg: WatchConfig) -> String {
            cfg.hostLabel.isEmpty ? ProcessInfo.processInfo.hostName : cfg.hostLabel
        }
    }
