    ```swift
    import Foundation
    import CoreServices

    /// Swift 6 strict-concurrency-safe wrapper around FSEventStream.
    /// Use a CallbackBox to bridge C-context-pointer to a Swift closure
    /// (Pitfall 4 mitigation — passing actor `self` directly triggers Sendability errors).
    public final class FSEventBridge: @unchecked Sendable {

        private var streamRef: FSEventStreamRef?
        private let queue: DispatchQueue
        private let box: CallbackBox

        // Hold the box's Unmanaged reference for the stream's lifetime.
        private var boxRetain: Unmanaged<CallbackBox>?

        public init(callback: @escaping ([String]) -> Void) {
            self.queue = DispatchQueue(label: "com.morrillholdings.vigil.watch.fsevent")
            self.box = CallbackBox(callback: callback)
        }

        /// Start watching `paths`. Returns true on success, false if FSEventStream creation fails.
        public func start(paths: [String]) -> Bool {
            guard streamRef == nil else { return true }   // already running

            let flags = FSEventStreamCreateFlags(
                kFSEventStreamCreateFlagFileEvents |
                kFSEventStreamCreateFlagNoDefer |
                kFSEventStreamCreateFlagWatchRoot
            )

            let retained = Unmanaged.passRetained(box)
            self.boxRetain = retained
            var context = FSEventStreamContext(
                version: 0,
                info: retained.toOpaque(),
                retain: nil,
                release: nil,
                copyDescription: nil
            )

            // C callback signature: (FSEventStreamRef, info pointer, numEvents, eventPaths, eventFlags, eventIds) -> Void
            let streamCallback: FSEventStreamCallback = { _, info, numEvents, eventPaths, _, _ in
                guard let info else { return }
                let box = Unmanaged<CallbackBox>.fromOpaque(info).takeUnretainedValue()
                let pathsArray = unsafeBitCast(eventPaths, to: NSArray.self) as? [String] ?? []
                box.callback(pathsArray)
            }

            let cfPaths = paths as CFArray
            guard let stream = FSEventStreamCreate(
                kCFAllocatorDefault,
                streamCallback,
                &context,
                cfPaths,
                FSEventStreamEventId(kFSEventStreamEventIdSinceNow),
                0.5,           // 0.5s latency — RESEARCH.md §1 "trailing-edge for tail-following"
                flags
            ) else {
                boxRetain?.release()
                boxRetain = nil
                return false
            }

            FSEventStreamSetDispatchQueue(stream, queue)
            FSEventStreamStart(stream)
            self.streamRef = stream
            return true
        }

        public func stop() {
            guard let stream = streamRef else { return }
            FSEventStreamStop(stream)
            FSEventStreamInvalidate(stream)
            FSEventStreamRelease(stream)
            boxRetain?.release()
            self.streamRef = nil
            self.boxRetain = nil
        }

        deinit {
            // Best-effort cleanup if start() was called and stop() wasn't.
            if let stream = streamRef {
                FSEventStreamStop(stream)
                FSEventStreamInvalidate(stream)
                FSEventStreamRelease(stream)
            }
            boxRetain?.release()
        }
    }

    /// Holds the Swift closure across the C-pointer boundary.
    private final class CallbackBox {
        let callback: ([String]) -> Void
        init(callback: @escaping ([String]) -> Void) { self.callback = callback }
    }
    ```

    Notes:
    - `@unchecked Sendable` on FSEventBridge: the streamRef is only mutated from start()/stop()/deinit which the daemon's main runloop serializes. The C callback dispatches through the closure to the actor.
    - `0.5` is the latency in seconds — trailing-edge per RESEARCH.md §1.
    - On macOS APFS, FSEventStream may receive a `kFSEventStreamEventFlagMustScanSubDirs` event under heavy load. We don't handle it explicitly here — the watcher (Task 8.2) re-reads from stored offsets on every callback, so a "scan subdirs" event simply triggers a full re-poll of all known files. Phase 123 can add explicit handling if needed.

    ```swift
    import Foundation

    /// Receives file-modification notifications from FSEventBridge, performs tail-cursor reads,
    /// parses lines, persists offsets, and dispatches ParsedLine values to a caller-provided handler.
    /// Plan 09 wires the handler to SessionStateRegistry → SessionState.process → EmitterActor.enqueue.
    public actor WatcherActor {

        private let stateStore: StateStore
        private let lineHandler: @Sendable (ParsedLine) async -> Void
        private var partialBuffers: [String: Data] = [:]    // path → bytes after last \n

        public init(stateStore: StateStore,
                    lineHandler: @escaping @Sendable (ParsedLine) async -> Void) {
            self.stateStore = stateStore
            self.lineHandler = lineHandler
        }

        // MARK: - Public surface

        /// Enumerate all <namespace>/*.jsonl files under projects_dir at startup, kick off initial reads.
        public func bootstrap(projectsDir: String) async {
            let expanded = ConfigLoader.expandHome(projectsDir)
            let baseURL = URL(fileURLWithPath: expanded, isDirectory: true)
            let fm = FileManager.default

            // Ensure watch root exists (FSEventStream is unreliable on non-existent paths — RESEARCH.md §1).
            if !fm.fileExists(atPath: baseURL.path) {
                logWarn("projects_dir does not exist yet: \(baseURL.path) — Claude Code will create it on first session")
                try? fm.createDirectory(at: baseURL, withIntermediateDirectories: true)
            }

            // Enumerate all <namespace>/ subdirectories.
            let subdirs: [URL]
            do {
                subdirs = try fm.contentsOfDirectory(at: baseURL,
                    includingPropertiesForKeys: [.isDirectoryKey],
                    options: [.skipsHiddenFiles])
            } catch {
                logWarn("could not enumerate \(baseURL.path): \(error.localizedDescription)")
                return
            }

            for subdir in subdirs {
                var isDir: ObjCBool = false
                guard fm.fileExists(atPath: subdir.path, isDirectory: &isDir), isDir.boolValue else { continue }
                let files = (try? fm.contentsOfDirectory(at: subdir, includingPropertiesForKeys: nil)) ?? []
                for file in files where file.pathExtension == "jsonl" {
                    await readNewBytes(filePath: file.path)
                }
            }
        }

        /// Called by FSEventBridge for every modified file path. Reads new bytes from stored offset.
        public func handleFileEvent(filePath: String) async {
            // Filter: only JSONL files. FSEventStream with kFSEventStreamCreateFlagFileEvents
            // may emit events for non-JSONL files in the same dir; ignore.
            guard filePath.hasSuffix(".jsonl") else { return }
            await readNewBytes(filePath: filePath)
        }

        // MARK: - Tail-cursor read

        /// Read everything from currentOffset to EOF. Split on \n; preserve any trailing partial line in
        /// partialBuffers[filePath]; only advance offset past complete \n-terminated bytes.
        /// For each complete line: parse → handler. Update offsets.json after each batch.
        private func readNewBytes(filePath: String) async {
            let currentOffset = await stateStore.getOffset(filePath: filePath)
            guard let fh = FileHandle(forReadingAtPath: filePath) else {
                // File may have been deleted between FSEvent fire and our read — silently ignore.
                return
            }
            defer { try? fh.close() }
            do {
                try fh.seek(toOffset: currentOffset)
            } catch {
                logWarn("seek failed on \(filePath) at offset \(currentOffset): \(error.localizedDescription)")
                return
            }
            guard let newData = try? fh.readToEnd(), !newData.isEmpty else {
                return
            }

            // Prepend any partial buffer from the previous read.
            var buffer: Data
            if let partial = partialBuffers[filePath] {
                buffer = partial + newData
            } else {
                buffer = newData
            }
            // The "anchor" offset: where buffer starts in the file = currentOffset minus partial-buffer size.
            let anchorOffset: UInt64 = currentOffset - UInt64(partialBuffers[filePath]?.count ?? 0)

            // Walk buffer, splitting on \n.
            var lineStart = 0
            var consumedUpTo = 0
            var i = 0
            while i < buffer.count {
                if buffer[i] == UInt8(ascii: "\n") {
                    let lineRange = lineStart..<i
                    let lineBytes = buffer[lineRange]
                    if !lineBytes.isEmpty,
                       let lineStr = String(data: lineBytes, encoding: .utf8) {
                        let lineByteOffset = anchorOffset + UInt64(lineStart)
                        await processLine(lineStr, byteOffset: lineByteOffset, filePath: filePath)
                    }
                    lineStart = i + 1
                    consumedUpTo = i + 1
                }
                i += 1
            }

            // Anything after the last \n is a partial line — buffer for next call.
            if lineStart < buffer.count {
                partialBuffers[filePath] = buffer[lineStart..<buffer.count]
            } else {
                partialBuffers[filePath] = nil
            }

            // Advance offset to consumedUpTo (only past complete lines).
            let newOffset = anchorOffset + UInt64(consumedUpTo)
            await stateStore.updateOffset(filePath: filePath, offset: newOffset)
            try? await stateStore.atomicSave()
        }

        /// Parse one complete line. On JSON parse failure → log + DO NOT EMIT (offset has already advanced
        /// past the line; per CONTEXT.md "do NOT advance the offset past the bad line" we have a slight
        /// trade-off here — we DO advance the offset past complete \n-terminated lines unconditionally,
        /// because partial-line detection drives offset advancement, not parse success. CONTEXT.md's
        /// guidance applies to BAD READS (incomplete data, NOT complete bytes that fail to parse).
        /// Bad JSON in a complete \n-terminated line is logged and skipped — re-attempting would be a no-op.)
        private func processLine(_ line: String, byteOffset: UInt64, filePath: String) async {
            guard let parsed = parseJSONLLine(line, byteOffset: byteOffset) else {
                logWarn("JSON parse failed at offset \(byteOffset) in \(filePath) — skipping (offset already advanced past complete \\n-terminated line)")
                return
            }
            // Non-spec lines: dispatch nothing to the handler; caller (Plan 09) will gate on lineType.
            await lineHandler(parsed)
        }

        // MARK: - Test inspection

        public func partialBufferSize(filePath: String) -> Int {
            partialBuffers[filePath]?.count ?? 0
        }
    }
    ```

    Important note on the "partial line" semantic — re-read the action body comment carefully and follow it. The original CONTEXT.md guidance "Skip and log JSON-parse failures without advancing offset" is about INCOMPLETE bytes (no `\n` yet) — Plan 02 ParserTests already covers bad-JSON returns nil. In WatcherActor, "advance only past complete `\n`-terminated lines" is the load-bearing invariant — bad JSON inside a complete line is logged and effectively dropped (re-reading wouldn't help; the bytes are intact and they're still bad JSON).

    The `anchorOffset` mechanic is critical for restart-replay safety:
    - On first read of a 1KB file: currentOffset=0, partial=nil → anchor=0, walk bytes, last `\n` at byte 990 → newOffset=990, partial=buffer[990..<1000].
    - File grows; FSEvent fires; second read: currentOffset=990, partial=10 bytes → re-seek to 990, read new bytes (50 more), prepend partial → buffer=60 bytes, anchor=990-10=980. Walk; last `\n` at buffer[55]+= → newOffset=anchor+56=1036.
    - Daemon restart between iterations: StateStore offset=990, no partial buffer in memory → next read starts at 990, partial=nil, no replay.
