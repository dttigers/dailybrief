    Implementation outline (follow CONTEXT.md §"JSONL parsing robustness" verbatim):

    ```swift
    import Foundation

    /// 7 known non-spec line types per Phase 120 README §"line-type noise filtering".
    /// Parser advances offset for these but emits ParsedLine.LineType.nonSpec so caller skips emission.
    public let nonSpecLineTypes: Set<String> = [
        "attachment",
        "queue-operation",
        "file-history-snapshot",
        "last-prompt",
        "ai-title",
        "summary",
        "system",
    ]

    /// Parse one complete `\n`-terminated JSONL line into a ParsedLine.
    /// Returns nil ONLY on JSON parse failure — caller MUST NOT advance offset past nil-returning lines
    /// (per CONTEXT.md "Skip and log any line that fails JSON parse — do NOT advance the offset").
    /// Returns ParsedLine with .nonSpec for the 7 known non-spec types — caller DOES advance offset.
    public func parseJSONLLine(_ line: String, byteOffset: UInt64) -> ParsedLine? {
        guard let data = line.data(using: .utf8),
              let obj = try? JSONSerialization.jsonObject(with: data) as? [String: Any] else {
            return nil
        }

        let typeStr = obj["type"] as? String ?? ""
        let sessionId = obj["sessionId"] as? String ?? ""
        let timestamp = parseISO8601(obj["timestamp"] as? String) ?? Date.distantPast
        let cwd = obj["cwd"] as? String
        let permissionMode = obj["permissionMode"] as? String

        // Non-spec lines: emit ParsedLine.nonSpec with best-effort metadata (caller advances offset, no event).
        if nonSpecLineTypes.contains(typeStr) {
            return ParsedLine(
                sessionId: sessionId, lineType: .nonSpec, timestamp: timestamp,
                byteOffset: byteOffset, cwd: cwd, permissionMode: permissionMode,
                stopReason: nil, toolUseIds: [], toolResultErrors: [], assistantTextBlocks: []
            )
        }

        // Spec lines: user / assistant.
        let lineType: ParsedLine.LineType
        switch typeStr {
        case "user":      lineType = .user
        case "assistant": lineType = .assistant
        default:          return nil    // unknown type — neither spec nor non-spec; skip-with-log at caller
        }

        // Inspect message.content[] for tool_use / tool_result / text blocks.
        var toolUseIds: [String] = []
        var toolResultErrors: [String] = []
        var assistantTextBlocks: [String] = []
        var stopReason: String? = nil

        if let message = obj["message"] as? [String: Any] {
            stopReason = message["stop_reason"] as? String
            if let content = message["content"] as? [[String: Any]] {
                for block in content {
                    let bType = block["type"] as? String ?? ""
                    switch bType {
                    case "tool_use":
                        // assistant lines: $.message.content[?(@.type=="tool_use")].id
                        if let id = block["id"] as? String { toolUseIds.append(id) }
                    case "tool_result":
                        // user lines: $.message.content[?(@.type=="tool_result" && @.is_error==true)].tool_use_id
                        if (block["is_error"] as? Bool) == true,
                           let tid = block["tool_use_id"] as? String {
                            toolResultErrors.append(tid)
                        }
                    case "text":
                        if lineType == .assistant, let txt = block["text"] as? String {
                            assistantTextBlocks.append(txt)
                        }
                    default: break
                    }
                }
            } else if let contentStr = message["content"] as? String, lineType == .user {
                // User lines occasionally carry $.message.content as a plain string (Phase 120 README §milestone row).
                // Surface as a single text block so the milestone matcher in Plan 03 can scan it.
                assistantTextBlocks.append(contentStr)
            }
        }

        return ParsedLine(
            sessionId: sessionId, lineType: lineType, timestamp: timestamp,
            byteOffset: byteOffset, cwd: cwd, permissionMode: permissionMode,
            stopReason: stopReason, toolUseIds: toolUseIds,
            toolResultErrors: toolResultErrors, assistantTextBlocks: assistantTextBlocks
        )
    }

    /// ISO-8601 parser. JSONL timestamps look like "2026-05-08T18:08:24.330Z".
    private func parseISO8601(_ s: String?) -> Date? {
        guard let s else { return nil }
        let f = ISO8601DateFormatter()
        f.formatOptions = [.withInternetDateTime, .withFractionalSeconds]
        if let d = f.date(from: s) { return d }
        f.formatOptions = [.withInternetDateTime]
        return f.date(from: s)
    }
    ```

    Key invariants to preserve verbatim:
    - Phase 120 README level-shift correction: top-level `tool_use` / `tool_result` types do NOT exist; only inner `$.message.content[].type` discriminates them.
    - Bad JSON → return nil (caller does NOT advance offset).
    - Unknown `$.type` (neither spec nor non-spec) → return nil (caller logs; offset still does not advance).
    - Non-spec types → return ParsedLine with `.nonSpec` (caller DOES advance offset, does NOT emit).
