    ```swift
    import Foundation

    /// On-disk schema for offsets.json (schema_version=2 per D-09).
    /// Hand-rolled Codable to control field naming exactly (snake_case keys, JSON-friendly).
    public struct StateFile: Codable, Sendable {
        public var schemaVersion: Int                                   // always 2 for Phase 122
        public var offsets: [String: UInt64]                            // <abs-file-path> → byteOffset
        public var milestonesEmitted: [String: [MilestoneRecord]]       // <sessionId> → records

        public enum CodingKeys: String, CodingKey {
            case schemaVersion = "schema_version"
            case offsets
            case milestonesEmitted = "milestones_emitted"
        }

        public init(schemaVersion: Int = 2,
                    offsets: [String: UInt64] = [:],
                    milestonesEmitted: [String: [MilestoneRecord]] = [:]) {
            self.schemaVersion = schemaVersion
            self.offsets = offsets
            self.milestonesEmitted = milestonesEmitted
        }
    }

    public enum StateStoreError: Error {
        case cannotCreateDirectory(String)
        case decodeFailed(Error)
        case schemaMismatch(found: Int)
        case writeFailed(String)
    }

    /// Single owner of `~/Library/Application Support/vigil-watch/offsets.json`.
    /// Atomic save: write-to-temp + F_FULLFSYNC + rename(2). macOS-specific crash-safe pattern per RESEARCH.md §6.
    public actor StateStore {

        public static let defaultStatePath: String = {
            let home = NSHomeDirectory()
            return "\(home)/Library/Application Support/vigil-watch/offsets.json"
        }()

        public static let sessionGCThresholdSeconds: TimeInterval = 24 * 60 * 60   // 24h per D-09

        private let path: URL
        private var state: StateFile

        public init(path: String = defaultStatePath, initialState: StateFile? = nil) {
            self.path = URL(fileURLWithPath: path)
            self.state = initialState ?? StateFile()
        }

        // MARK: - Lifecycle

        /// Load existing state from disk if present; create empty if absent.
        /// On schema_version mismatch (legacy v1), throws — daemon should log + exit non-zero
        /// (no migration support in Phase 122; v2 is the first real schema).
        /// Lazy-GC sessions older than 24h on load per D-09.
        public func loadOrCreate() throws {
            let fm = FileManager.default
            let dir = path.deletingLastPathComponent()
            if !fm.fileExists(atPath: dir.path) {
                do {
                    try fm.createDirectory(at: dir, withIntermediateDirectories: true)
                } catch {
                    throw StateStoreError.cannotCreateDirectory(dir.path)
                }
            }
            guard fm.fileExists(atPath: path.path) else {
                self.state = StateFile()
                try atomicSaveLocked()
                return
            }
            let data: Data
            do { data = try Data(contentsOf: path) }
            catch { throw StateStoreError.decodeFailed(error) }
            let decoded: StateFile
            do { decoded = try JSONDecoder().decode(StateFile.self, from: data) }
            catch { throw StateStoreError.decodeFailed(error) }
            guard decoded.schemaVersion == 2 else {
                throw StateStoreError.schemaMismatch(found: decoded.schemaVersion)
            }
            self.state = gcOldSessions(decoded)
        }

        // MARK: - Mutations

        public func updateOffset(filePath: String, offset: UInt64) {
            state.offsets[filePath] = offset
        }

        public func getOffset(filePath: String) -> UInt64 {
            state.offsets[filePath] ?? 0
        }

        public func recordMilestone(sessionId: String, pattern: String, firstMatchOffset: UInt64,
                                    emittedAt: String = ISO8601DateFormatter().string(from: Date())) {
            var existing = state.milestonesEmitted[sessionId] ?? []
            // Skip if already recorded (defensive — caller's matcher should have already deduped).
            if !existing.contains(where: { $0.pattern == pattern }) {
                existing.append(MilestoneRecord(pattern: pattern,
                                                firstMatchOffset: firstMatchOffset,
                                                emittedAt: emittedAt))
                state.milestonesEmitted[sessionId] = existing
            }
        }

        public func priorEmissions() -> [String: [MilestoneRecord]] {
            return state.milestonesEmitted
        }

        public func snapshotForTesting() -> StateFile { state }

        // MARK: - Atomic save

        /// Write state to disk atomically: temp + F_FULLFSYNC + rename.
        /// Caller invokes after each batch of mutations or periodically.
        public func atomicSave() throws {
            try atomicSaveLocked()
        }

        private func atomicSaveLocked() throws {
            let dir = path.deletingLastPathComponent()
            let tmp = dir.appendingPathComponent(path.lastPathComponent + ".tmp")
            let encoder = JSONEncoder()
            encoder.outputFormatting = [.prettyPrinted, .sortedKeys]
            let data: Data
            do { data = try encoder.encode(state) }
            catch { throw StateStoreError.writeFailed("encode: \(error)") }

            // Write temp file.
            do { try data.write(to: tmp) }
            catch { throw StateStoreError.writeFailed("write temp: \(error)") }

            // F_FULLFSYNC the temp file (RESEARCH.md §6 — macOS-specific stronger fsync).
            let fd = open(tmp.path, O_RDONLY)
            if fd != -1 {
                _ = fcntl(fd, F_FULLFSYNC)
                close(fd)
            }
            // Atomic rename (same-filesystem rename(2) is atomic on macOS APFS / HFS+).
            do {
                if FileManager.default.fileExists(atPath: path.path) {
                    _ = try? FileManager.default.removeItem(at: path)
                }
                try FileManager.default.moveItem(at: tmp, to: path)
            } catch {
                throw StateStoreError.writeFailed("rename: \(error)")
            }
        }

        // MARK: - GC (D-09)

        private func gcOldSessions(_ s: StateFile) -> StateFile {
            // Drop entries from milestones_emitted whose newest emittedAt is older than 24h.
            var out = s
            let cutoff = Date().addingTimeInterval(-Self.sessionGCThresholdSeconds)
            let f = ISO8601DateFormatter()
            f.formatOptions = [.withInternetDateTime, .withFractionalSeconds]
            out.milestonesEmitted = s.milestonesEmitted.filter { _, records in
                let newest = records.compactMap { rec -> Date? in
                    if let d = f.date(from: rec.emittedAt) { return d }
                    var f2 = ISO8601DateFormatter(); f2.formatOptions = [.withInternetDateTime]
                    return f2.date(from: rec.emittedAt)
                }.max() ?? Date.distantPast
                return newest >= cutoff
            }
            return out
        }
    }
    ```

    Notes:
    - `loadOrCreate` creates the directory at `~/Library/Application Support/vigil-watch/` if absent.
    - On legacy schema (no `schema_version` field, or != 2), throw rather than silently migrate. Phase 122 is the first phase to write this file — there is no v1 in the wild.
    - `atomicSave()` is the only mutator that touches disk. All other methods are pure in-memory updates.
    - F_FULLFSYNC ignores errors — best-effort. The atomic rename guarantees consistency even if F_FULLFSYNC didn't flush; F_FULLFSYNC is a durability hardener, not correctness boundary.
