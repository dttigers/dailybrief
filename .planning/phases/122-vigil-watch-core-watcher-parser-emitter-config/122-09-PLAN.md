    ```swift
    import Foundation

    /// Owns per-session SessionState actors. Get-or-create on first observation of a sessionId.
    public actor SessionRegistry {

        private var sessions: [String: SessionState] = [:]
        private let label: String
        private let host: String

        public init(label: String, host: String) {
            self.label = label
            self.host = host
        }

        /// Returns existing SessionState or creates a new one. Phase 122 does NOT garbage-collect
        /// in-memory SessionState (Phase 123's 24h soak will surface if memory growth becomes an issue).
        public func getOrCreate(sessionId: String) -> SessionState {
            if let existing = sessions[sessionId] { return existing }
            let new = SessionState(sessionId: sessionId, label: label, host: host)
            sessions[sessionId] = new
            return new
        }

        public func allSessions() -> [SessionState] {
            return Array(sessions.values)
        }

        public func count() -> Int { sessions.count }
    }
    ```

    ```swift
    import Foundation

    /// Held for the lifetime of the process to keep the DispatchSource alive.
    nonisolated(unsafe) private var _sigtermSource: DispatchSourceSignal?
    nonisolated(unsafe) private var _sigintSource: DispatchSourceSignal?

    /// Install graceful-shutdown handler for SIGTERM (and SIGINT for Ctrl-C in foreground).
    /// Per RESEARCH.md §8: signal() directly is unsafe in Swift; use DispatchSource.makeSignalSource.
    /// Pattern: signal(SIG, SIG_IGN) first so GCD takes over delivery; then makeSignalSource.
    /// On SIGTERM: drain queue with 5s deadline, log, exit(0).
    public func installSIGTERMHandler(emitter: EmitterActor) {
        signal(SIGTERM, SIG_IGN)
        signal(SIGINT, SIG_IGN)

        let term = DispatchSource.makeSignalSource(signal: SIGTERM, queue: DispatchQueue.global())
        term.setEventHandler {
            Task {
                logInfo("SIGTERM received — draining queue (5s deadline)")
                _ = await emitter.drain(deadlineSeconds: 5.0)
                logInfo("vigil-watch exiting")
                exit(0)
            }
        }
        term.resume()
        _sigtermSource = term

        let intr = DispatchSource.makeSignalSource(signal: SIGINT, queue: DispatchQueue.global())
        intr.setEventHandler {
            Task {
                logInfo("SIGINT received — draining queue (5s deadline)")
                _ = await emitter.drain(deadlineSeconds: 5.0)
                logInfo("vigil-watch exiting")
                exit(0)
            }
        }
        intr.resume()
        _sigintSource = intr
    }
    ```

    Notes:
    - `nonisolated(unsafe)` is required for the global DispatchSource holders under Swift 6 strict concurrency. The values are written exactly once during startup; reads happen only on the global dispatch queue inside the handlers.
    - Both SIGTERM and SIGINT handled identically — Phase 123 launchd delivers SIGTERM; foreground Ctrl-C delivers SIGINT.

    ```swift
        public let config: WatchConfig
        public let emitter: EmitterActor
        public let stateStore: StateStore

        public let registry: SessionRegistry
        public let milestone: MilestoneMatcher

        public init(config: WatchConfig) async throws {
            self.config = config
            // 1. State store — load or create offsets.json schema_version=2.
            let store = StateStore()
            try await store.loadOrCreate()
            self.stateStore = store

            // 2. Milestone matcher — hydrate from prior emissions for D-08 restart safety.
            let prior = await store.priorEmissions()
            self.milestone = MilestoneMatcher(patterns: config.milestonePatterns, priorEmissions: prior)

            // 3. Emitter — owns retry/backoff queue. Quarantine if api_key empty.
            self.emitter = EmitterActor(config: config)

            // 4. Session registry — get-or-create per sessionId.
            let resolvedHost = ConfigLoader.resolveHost(config)
            // Label: use the last component of cwd from each line at run time. Bootstrap label = "" (set per-line).

            // 5. Watcher — closure capture for line dispatch (set up below).
            // We need a forward reference, so declare a placeholder and re-bind via closure.
            let registryRef = self.registry
            let emitterRef = self.emitter
            let milestoneRef = self.milestone
            let storeRef = self.stateStore

            self.watcher = WatcherActor(stateStore: store, lineHandler: { [weak emitterRef, weak storeRef] line in
                guard let emitter = emitterRef, let store = storeRef else { return }

                // Skip non-spec lines silently.
                if line.lineType == .nonSpec { return }

                let session = await registryRef.getOrCreate(sessionId: line.sessionId)
                // Dispatch the line — captures any IMMEDIATE-fire events (task_failed).
                let immediateEvents = await session.process(line: line)
                for ev in immediateEvents {
                    await emitter.enqueue(ev)
                }

                // Milestone matching: only assistant lines, only first-time matches per (sessionId, pattern).
                if line.lineType == .assistant {
                    let hits = await milestoneRef.scanForFirstTimeMatches(line: line)
                    for compiled in hits {
                        let f = ISO8601DateFormatter()
                        f.formatOptions = [.withInternetDateTime, .withFractionalSeconds]
                        let payload = VigilPayload(
                            sessionId: line.sessionId,
                            event: VigilEvent.milestone.rawValue,
                            message: "milestone: \(compiled.raw)",
                            timestamp: f.string(from: line.timestamp),
                            label: line.cwd.flatMap { URL(fileURLWithPath: $0).lastPathComponent } ?? "",
                            host: ConfigLoader.resolveHost((await emitter).config) ,
                            // ↑ slight hack: emitter exposes config via a public read; if it doesn't,
                            //   use config captured in closure. See note below.
                            exitCode: nil,
                            clientEventId: makeClientEventId(sessionId: line.sessionId,
                                                             byteOffset: line.byteOffset,
                                                             eventType: VigilEvent.milestone.rawValue)
                        )
                        await emitter.enqueue(payload)
                        await milestoneRef.recordEmission(sessionId: line.sessionId, pattern: compiled.raw)
                        await store.recordMilestone(sessionId: line.sessionId,
                                                    pattern: compiled.raw,
                                                    firstMatchOffset: line.byteOffset)
                    }
                    if !hits.isEmpty {
                        try? await store.atomicSave()
                    }
                }
            })
        }

        // Resolve a small awkwardness: the line-handler closure above references `(await emitter).config`
        // which doesn't compile (EmitterActor.config is private). Refactor by capturing host directly:
        // (the executor should follow this pattern instead — capture `let resolvedHost` and use it inline)

        public func start() async {
            logInfo("vigil-watch starting — projects_dir=\(ConfigLoader.expandHome(config.projectsDir)) host=\(ConfigLoader.resolveHost(config))")

            if config.apiKey.isEmpty {
                logWarn("api_key missing in watch.toml AND VIGIL_API_KEY env var — running in QUARANTINE mode (events queued, not posted). Set api_key and restart.")
            }

            // 1. Bootstrap watcher: enumerate existing JSONL files, read from stored offsets.
            await watcher.bootstrap(projectsDir: config.projectsDir)

            // 2. Start FSEventStream watching the projects_dir.
            let watchPath = ConfigLoader.expandHome(config.projectsDir)
            let watcherRef = self.watcher
            let bridge = FSEventBridge(callback: { paths in
                Task {
                    for path in paths {
                        await watcherRef.handleFileEvent(filePath: path)
                    }
                }
            })
            if !bridge.start(paths: [watchPath]) {
                logError("failed to start FSEventStream on \(watchPath)")
                return
            }
            self.bridge = bridge

            // 3. Start emitter flush loop.
            await emitter.startFlushLoop()

            // 4. Start the 1Hz session evaluation tick (timer-driven events: needs_input/task_complete/heartbeat).
            let registryRef = self.registry
            let emitterRef = self.emitter
            let cfg = config
            evaluationTask = Task {
                while !Task.isCancelled {
                    try? await Task.sleep(for: .seconds(1))
                    if Task.isCancelled { break }
                    let sessions = await registryRef.allSessions()
                    for s in sessions {
                        let events = await s.evaluate(now: Date(), config: cfg)
                        for e in events { await emitterRef.enqueue(e) }
                    }
                }
            }

            logInfo("vigil-watch started — watching \(watchPath)")
        }

        public func stop() async {
            evaluationTask?.cancel()
            evaluationTask = nil
            bridge?.stop()
            bridge = nil
            await emitter.stopFlushLoop()
        }
    }
    ```
