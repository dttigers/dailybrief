    ```swift
    import Foundation

    /// Persistable record of one milestone emission. Plan 06 StateStore writes/reads these
    /// to/from offsets.json schema_version=2 → milestones_emitted[sessionId][N].
    public struct MilestoneRecord: Codable, Equatable, Sendable {
        public let pattern: String          // raw pattern as stored in watch.toml (without (?i) prefix)
        public let firstMatchOffset: UInt64 // byteOffset of the JSONL line where first match landed
        public let emittedAt: String        // ISO-8601

        public init(pattern: String, firstMatchOffset: UInt64, emittedAt: String) {
            self.pattern = pattern
            self.firstMatchOffset = firstMatchOffset
            self.emittedAt = emittedAt
        }
    }

    /// Match-once-per-session-per-pattern matcher per D-07.
    /// Compiles all WatchConfig.milestonePatterns at construction time with implicit (?i) prepend (D-10).
    /// Patterns that fail compilation are logged and skipped — daemon continues running (T-122-03 mitigation).
    public actor MilestoneMatcher {

        public struct Compiled: Sendable {
            public let raw: String           // raw pattern text from watch.toml
            public let regex: NSRegularExpression
        }

        private var compiled: [Compiled]
        private var emittedBySession: [String: Set<String>]  // sessionId → set of raw patterns

        public init(patterns: [String], priorEmissions: [String: [MilestoneRecord]] = [:]) {
            // Compile each pattern with (?i) prepended unless it already contains an explicit (?i) or (?-i) flag.
            var out: [Compiled] = []
            for raw in patterns {
                let withFlag: String = {
                    // If pattern already opens with `(?` flag group, leave it alone — user opt-out per D-10.
                    if raw.hasPrefix("(?") { return raw }
                    return "(?i)" + raw
                }()
                do {
                    let r = try NSRegularExpression(pattern: withFlag)
                    out.append(Compiled(raw: raw, regex: r))
                } catch {
                    logWarn("milestone pattern compile failed — skipping: \(raw) — \(error.localizedDescription)")
                }
            }
            self.compiled = out

            // Hydrate dedupe state from prior emissions (D-08 — survives daemon restart).
            var seed: [String: Set<String>] = [:]
            for (sessionId, records) in priorEmissions {
                seed[sessionId] = Set(records.map(\.pattern))
            }
            self.emittedBySession = seed
        }

        /// Scan all assistant text blocks of a parsed line. Returns the SUBSET of compiled patterns
        /// whose regex matches AND has not yet been emitted for this sessionId.
        /// CALLER is responsible for invoking `recordEmission(...)` after the milestone event is enqueued.
        public func scanForFirstTimeMatches(line: ParsedLine) -> [Compiled] {
            guard line.lineType == .assistant, !line.assistantTextBlocks.isEmpty else { return [] }
            let alreadyEmitted = emittedBySession[line.sessionId] ?? []
            var hits: [Compiled] = []
            for c in compiled where !alreadyEmitted.contains(c.raw) {
                let matched = line.assistantTextBlocks.contains { text in
                    let nsRange = NSRange(text.startIndex..., in: text)
                    return c.regex.firstMatch(in: text, options: [], range: nsRange) != nil
                }
                if matched { hits.append(c) }
            }
            return hits
        }

        /// Record that the given pattern has now been emitted for this session.
        /// Plan 06 StateStore mirrors this into offsets.json so a daemon restart preserves the dedupe.
        public func recordEmission(sessionId: String, pattern: String) {
            var set = emittedBySession[sessionId] ?? []
            set.insert(pattern)
            emittedBySession[sessionId] = set
        }

        /// Inspection — Plan 06 StateStore calls this to snapshot before atomic save.
        public func snapshot() -> [String: Set<String>] {
            return emittedBySession
        }

        /// Test/inspection helper.
        public func compiledCount() -> Int { compiled.count }
        public func hasEmitted(sessionId: String, pattern: String) -> Bool {
            (emittedBySession[sessionId] ?? []).contains(pattern)
        }
    }
    ```

    Design notes:
    - Patterns are stored RAW (without `(?i)` prefix) in `Compiled.raw` so the dedupe key matches what's in watch.toml — when Plan 06 persists/restores, the keys round-trip cleanly.
    - User opt-out for case-insensitivity: if a pattern starts with `(?` (e.g., `(?-i)CaseSensitive`), the matcher leaves it alone. Per D-10.
    - `scanForFirstTimeMatches` returns the COMPILED patterns matched (not booleans) so the caller can both build the milestone payload's `message` field AND know which raw pattern to record.
    - All NSRegularExpression instances are constructed once at init time. The actor isolates concurrent reads safely.
