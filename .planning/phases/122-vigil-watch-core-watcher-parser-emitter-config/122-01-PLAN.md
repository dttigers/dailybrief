    ```swift
    import CryptoKit
    import Foundation

    // D-02: rule version is a Swift const, NEVER a DB column.
    // Bump only for deliberate operator-initiated backfills (rewind offsets, replay, revert).
    public let ruleVersion = "v1"

    /// D-01: deterministic client_event_id derived from
    ///   sha256("\(sessionId)|\(byteOffset)|\(eventType)|\(ruleVersion)")
    /// Hex-encoded prefix-32 (16 bytes) formatted as UUID 8-4-4-4-12.
    /// Same input always produces same output → restart-replay safe via
    /// Postgres partial unique index (user_id, client_event_id) WHERE client_event_id IS NOT NULL.
    public func makeClientEventId(sessionId: String, byteOffset: UInt64, eventType: String) -> String {
        let input = "\(sessionId)|\(byteOffset)|\(eventType)|\(ruleVersion)"
        let digest = SHA256.hash(data: Data(input.utf8))
        let fullHex = digest.compactMap { String(format: "%02x", $0) }.joined()
        let h = String(fullHex.prefix(32))
        let i0 = h.index(h.startIndex, offsetBy: 8)
        let i1 = h.index(i0, offsetBy: 4)
        let i2 = h.index(i1, offsetBy: 4)
        let i3 = h.index(i2, offsetBy: 4)
        return "\(h[..<i0])-\(h[i0..<i1])-\(h[i1..<i2])-\(h[i3...])".replacingOccurrences(of: "", with: "") // placeholder — see action body
    }
    ```

    ```swift
    import XCTest
    @testable import VigilWatch

    final class DriftDetectorTests: XCTestCase {

        /// Phase 121 D-T2 carry-forward — locks the 5 VALID_EVENTS strings in
        /// vigil-core/src/routes/agent-events.ts byte-identical to VigilEvent.allCases.
        /// If vigil-core ever renames or reorders an event, this test fails immediately
        /// instead of vigil-watch silently posting events vigil-core 400-rejects.
        func testVigilEventCasesMatchAgentEventsTS() throws {
            let vigilCorePath = ProcessInfo.processInfo.environment["VIGIL_CORE_PATH"]
                ?? "\(NSHomeDirectory())/Desktop/Local AI/dailybrief/vigil-core"
            let tsFile = URL(fileURLWithPath: vigilCorePath)
                .appendingPathComponent("src/routes/agent-events.ts")

            guard FileManager.default.fileExists(atPath: tsFile.path) else {
                throw XCTSkip("VIGIL_CORE_PATH not pointing at a vigil-core checkout " +
                              "(\(tsFile.path) missing). Set VIGIL_CORE_PATH to skip-or-run.")
            }
            let source = try String(contentsOf: tsFile, encoding: .utf8)

            // Extract VALID_EVENTS array from TS source.
            let pattern = #"export const VALID_EVENTS = \[([\s\S]*?)\] as const"#
            let regex = try NSRegularExpression(pattern: pattern)
            let nsRange = NSRange(source.startIndex..., in: source)
            guard let match = regex.firstMatch(in: source, range: nsRange),
                  let bodyRange = Range(match.range(at: 1), in: source) else {
                XCTFail("Could not locate `export const VALID_EVENTS = [...] as const` in agent-events.ts")
                return
            }
            let body = String(source[bodyRange])

            // Pull each quoted string token out of the array body.
            let tsEvents: [String] = body
                .components(separatedBy: "\n")
                .compactMap { line in
                    let trimmed = line.trimmingCharacters(in: .whitespaces)
                    guard trimmed.hasPrefix("\"") else { return nil }
                    return trimmed.trimmingCharacters(in: CharacterSet(charactersIn: "\","))
                }

            let swiftEvents = VigilEvent.allCases.map { $0.rawValue }

            XCTAssertEqual(Set(swiftEvents), Set(tsEvents),
                "DRIFT DETECTED: VigilEvent.allCases.map(\\.rawValue) diverged from " +
                "VALID_EVENTS in vigil-core/src/routes/agent-events.ts. " +
                "Swift: \(swiftEvents). TS: \(tsEvents). " +
                "Either vigil-core renamed an event (update VigilEvent enum) " +
                "or vigil-watch's enum drifted (revert).")

            XCTAssertEqual(swiftEvents.count, 5, "Phase 121 D-A4 locked 5 events; count drift = redesign")
            XCTAssertEqual(tsEvents.count, 5, "vigil-core has != 5 events; check VALID_EVENTS in agent-events.ts")
        }
    }
    ```
