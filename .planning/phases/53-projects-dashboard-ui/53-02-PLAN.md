**File 3: `Sources/JarvisCore/Storage/ProjectsAPIStore.swift` (NEW)**

```swift
import Foundation

public actor ProjectsAPIStore: ProjectsRepository {
    private let client: VigilAPIClient

    public init(client: VigilAPIClient) {
        self.client = client
    }

    // MARK: - Wire DTO

    private struct APIProjectResponse: Decodable, Sendable {
        let id: Int64
        let name: String
        let description: String?
        let status: String?
        let createdAt: Date
        let updatedAt: Date
    }

    private func toProject(_ r: APIProjectResponse) -> Project {
        Project(
            id: r.id,
            name: r.name,
            description: r.description,
            status: r.status.flatMap { ProjectStatus(rawValue: $0) },
            createdAt: r.createdAt,
            updatedAt: r.updatedAt
        )
    }

    // MARK: - List

    public func listProjects() async throws -> [Project] {
        let response: [APIProjectResponse] = try await client.get(path: "/projects")
        return response.map(toProject)
    }

    // MARK: - Create

    private struct CreateProjectBody: Encodable {
        let name: String
        let description: String?
        let status: String?
    }

    public func createProject(
        name: String,
        description: String?,
        status: ProjectStatus?
    ) async throws -> Project {
        let body = CreateProjectBody(
            name: name,
            description: description,
            status: status?.rawValue
        )
        let r: APIProjectResponse = try await client.post(path: "/projects", body: body)
        return toProject(r)
    }

    // MARK: - Update

    private struct UpdateProjectBody: Encodable {
        let name: String?
        let description: String?
        let status: String?
    }

    public func updateProject(
        id: Int64,
        name: String?,
        description: String?,
        status: ProjectStatus?
    ) async throws -> Project {
        let body = UpdateProjectBody(
            name: name,
            description: description,
            status: status?.rawValue
        )
        let r: APIProjectResponse = try await client.patch(
            path: "/projects/\(id)",
            body: body
        )
        return toProject(r)
    }

    // MARK: - Delete

    public func deleteProject(id: Int64) async throws {
        try await client.delete(path: "/projects/\(id)")
    }
}
```

public func patch<T: Decodable, B: Encodable>(path: String, body: B) async throws -> T {

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Mac client → Railway vigil-core | All project/thought I/O crosses the TLS + bearer-auth boundary |
| JSON decoder → Project/Thought value types | Untrusted server JSON is decoded into Swift values; decoder must fail closed on shape drift |
| Swift call sites → AssignProjectBody wire format | `Optional.none` in Swift becomes JSON `null` — must only be used where "unassign" is the intended semantic |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-53-10 | Tampering | VigilAPIClient.patch — TLS + bearer auth | mitigate | Reuses existing `applyHeaders`, `perform`, and URLSession session from `put`. No new transport path. Pinned to the same baseURL. |
| T-53-11 | Tampering | Swift JSON encoding of `Optional.none` as `null` (Pitfall P-1) | mitigate | Dedicated `AssignProjectBody` struct used ONLY by `updateProjectId`. Explicitly NOT added to the shared `UpdateThoughtBody`. Code comment + plan acceptance criterion forbids mixing. |
| T-53-12 | Information disclosure | Bearer token leakage via logs | mitigate | VigilAPIClient's existing pattern does not log request bodies or Authorization headers. Not changed by this plan. |
| T-53-13 | Denial of service | Unbounded `limit` on fetchByProject/fetchUnassigned | mitigate | Both methods default `limit = 200`. Caller may override but the existing thoughts route already enforces its own max limit server-side. |
