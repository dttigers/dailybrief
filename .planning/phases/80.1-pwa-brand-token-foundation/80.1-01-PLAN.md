@import "tailwindcss";
@import url('https://fonts.googleapis.com/css2?family=Inter:wght@400;500&display=swap');

@theme {
  /* Font */
  --font-sans: 'Inter', ui-sans-serif, system-ui, sans-serif;

  /* Vigil Teal */
  --color-teal-50: #E1F5EE;
  --color-teal-100: #9FE1CB;
  --color-teal-400: #1D9E75;
  --color-teal-600: #0F6E56;

  --color-gray-50: #F1EFE8;
  --color-gray-100: #D3D1C7;
  --color-gray-400: #888780;
  --color-gray-900: #2C2C2A;
  --color-surface: #F9F9F8;

  /* Status Accents */
  --color-info-50: #E6F1FB;
  --color-info-400: #378ADD;
  --color-warning-50: #FAEEDA;
  --color-warning-400: #BA7517;
  --color-success-50: #EAF3DE;
  --color-success-400: #639922;
}
```

IMPORTANT: Do NOT use `--color-*: initial` anywhere. This would wipe all default Tailwind colors including `red-*` used for delete buttons and error states. Add tokens additively only.

<link rel="preconnect" href="https://fonts.gstatic.com" crossorigin>
<link href="https://fonts.googleapis.com/css2?family=Inter:wght@400;500&display=swap" rel="stylesheet">
```
