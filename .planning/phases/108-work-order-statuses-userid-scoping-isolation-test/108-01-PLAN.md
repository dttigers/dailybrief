DO $$

--> statement-breakpoint

ALTER TABLE "work_order_statuses" ALTER COLUMN "user_id" SET NOT NULL;
--> statement-breakpoint

-- ── Step 4: ADD FK to users(id) ON DELETE RESTRICT ────────────────────────
DO $$ BEGIN
