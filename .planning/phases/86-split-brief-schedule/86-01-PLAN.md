          function isValidTimezone(tz: unknown): tz is string {
            if (typeof tz !== "string" || tz.length === 0) return false;
            try {
              const resolved = new Intl.DateTimeFormat("en-US", { timeZone: tz }).resolvedOptions().timeZone;
