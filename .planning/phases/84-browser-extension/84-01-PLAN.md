Create `vigil-extension/popup.js` with the following exact logic. This is vanilla JS (per D-01), no modules, no imports. The file is loaded via a script tag in popup.html.

**Complete popup.js contents:**
```javascript
'use strict';

const STORAGE_KEY = 'vigil_api_key';
const API_BASE = 'https://api.vigilhub.io';

// --- DOM references ---
const setupView = document.getElementById('setup-view');
const captureView = document.getElementById('capture-view');
const apiKeyInput = document.getElementById('api-key-input');
const saveKeyBtn = document.getElementById('save-key-btn');
const setupError = document.getElementById('setup-error');
const setupLoading = document.getElementById('setup-loading');
const contentInput = document.getElementById('content-input');
const captureBtn = document.getElementById('capture-btn');
const captureError = document.getElementById('capture-error');
const captureSuccess = document.getElementById('capture-success');
const settingsBtn = document.getElementById('settings-btn');

// --- View management ---

function showView(view) {
  setupView.hidden = true;
  captureView.hidden = true;
  view.hidden = false;
}

// --- API key validation ---
// Uses /v1/summary (NOT /v1/health -- health returns 200 without auth)

async function validateApiKey(key) {
  try {
    const res = await fetch(`${API_BASE}/v1/summary`, {
      headers: { 'Authorization': `Bearer ${key}` },
    });
    return res.ok;
  } catch {
    return false;
  }
}

// --- Setup flow ---

saveKeyBtn.addEventListener('click', async () => {
  const key = apiKeyInput.value.trim();
  if (!key) {
    setupError.textContent = 'Please enter an API key.';
    setupError.hidden = false;
    return;
  }

  setupError.hidden = true;
  setupLoading.hidden = false;
  saveKeyBtn.disabled = true;

  const valid = await validateApiKey(key);

  setupLoading.hidden = true;
  saveKeyBtn.disabled = false;

  if (!valid) {
    setupError.textContent = 'Invalid API key. Check your key and try again.';
    setupError.hidden = false;
    return;
  }

  await chrome.storage.local.set({ [STORAGE_KEY]: key });
  await initCaptureView(key);
});

// Allow Enter key to submit on the API key input
apiKeyInput.addEventListener('keydown', (e) => {
  if (e.key === 'Enter') {
    saveKeyBtn.click();
  }
});

// --- Capture flow ---

async function initCaptureView(apiKey) {
  showView(captureView);
  captureError.hidden = true;
  captureSuccess.hidden = true;

  // Read active tab -- must happen during DOMContentLoaded (user gesture context)
  try {
    const [tab] = await chrome.tabs.query({ active: true, currentWindow: true });
    const title = tab.title ?? '';
    const url = tab.url ?? '';
    // Pre-fill: title on first line, URL on second line, blank line, cursor position for note
    contentInput.value = `${title}\n${url}\n\n`;
    // Place cursor at the end (after the blank line) for the user to type a note
    contentInput.focus();
    contentInput.selectionStart = contentInput.value.length;
    contentInput.selectionEnd = contentInput.value.length;
  } catch {
    // If tab query fails (e.g., chrome:// pages), leave empty
    contentInput.value = '';
    contentInput.focus();
  }

  // Wire capture button
  captureBtn.onclick = async () => {
    const content = contentInput.value.trim();
    if (!content) {
      captureError.textContent = 'Content cannot be empty.';
      captureError.hidden = false;
      return;
    }

    captureError.hidden = true;
    captureSuccess.hidden = true;
    captureBtn.disabled = true;
    captureBtn.textContent = 'Capturing...';

    try {
      const res = await fetch(`${API_BASE}/v1/thoughts`, {
        method: 'POST',
        headers: {
          'Authorization': `Bearer ${apiKey}`,
          'Content-Type': 'application/json',
        },
        body: JSON.stringify({ content: content, source: 'text' }),
      });

      if (!res.ok) {
        const status = res.status;
        if (status === 401) {
          captureError.textContent = 'API key rejected (401). Check your key in settings.';
        } else {
          captureError.textContent = `Capture failed (HTTP ${status}). Try again.`;
        }
        captureError.hidden = false;
        captureBtn.disabled = false;
        captureBtn.textContent = 'Capture';
        return;
      }

      // Success -- show confirmation
      captureSuccess.hidden = false;
      captureBtn.textContent = 'Capture';
      captureBtn.disabled = false;

      // Auto-close popup after 1.5 seconds on success
      setTimeout(() => window.close(), 1500);
    } catch (err) {
      captureError.textContent = `Network error: ${err.message}`;
      captureError.hidden = false;
      captureBtn.disabled = false;
      captureBtn.textContent = 'Capture';
    }
  };
}

// --- Settings button (switch back to setup view) ---

settingsBtn.addEventListener('click', async () => {
  const { [STORAGE_KEY]: currentKey } = await chrome.storage.local.get([STORAGE_KEY]);
  apiKeyInput.value = currentKey ?? '';
  showView(setupView);
});

// --- Init on popup open ---

document.addEventListener('DOMContentLoaded', async () => {
  const { [STORAGE_KEY]: apiKey } = await chrome.storage.local.get([STORAGE_KEY]);

  if (!apiKey) {
    showView(setupView);
  } else {
    await initCaptureView(apiKey);
  }
});
```

Key design decisions (Claude's discretion areas):
- **Content format:** Title on line 1, URL on line 2, blank line 3, then user note. Simple and readable.
- **Post-capture UX:** Success checkmark message + auto-close after 1.5s. Quick and non-blocking.
- **API key onboarding:** Inline in popup (not a separate options page). Lean for a single-setting extension.
- **No category dropdown:** Server default handles categorization. Extension is for quick capture only.
- **Settings gear button:** Allows changing API key without clearing storage manually.

Critical implementation notes:
- `source: 'text'` (NOT 'browser' -- would 400, per VALID_SOURCES in vigil-core)
- Key validation uses `/v1/summary` (NOT `/v1/health` -- health returns 200 without auth)
- `chrome.storage.local` (NOT sync -- sync is unencrypted and replicates credentials)
- `activeTab` permission (NOT `tabs` -- avoids "Read browsing history" warning)
- Tab query in DOMContentLoaded (popup open IS the user gesture granting activeTab access)
- All chrome.storage calls use `await` (async API, no synchronous read)

<threat_model>
## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Extension -> Vigil API | Bearer token crosses from local storage to HTTPS request |
| User input -> textarea | Page title pre-fills textarea; user edits before submitting |
| API key input -> storage | User-provided key stored in chrome.storage.local |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-84-01 | Information Disclosure | chrome.storage.local | accept | API key stored unencrypted at rest in local storage; acceptable for single-user tool -- key already stored in multiple locations (config.json, .env, plist per project memory). storage.local is NOT synced (unlike sync). Only this extension's code can read it. |
| T-84-02 | Spoofing | Bearer token in HTTPS request | mitigate | Always use HTTPS (host_permissions enforces https://api.vigilhub.io/*). Token never sent over HTTP. |
| T-84-03 | Tampering | Page title injection into textarea | mitigate | Title is set via `.value` property on textarea (NOT innerHTML). User sees and edits content before submitting. No XSS vector. |
| T-84-04 | Elevation of Privilege | CORS bypass | accept | host_permissions grants extension fetch access to api.vigilhub.io. This is the intended MV3 pattern. Server validates Bearer auth independently. |
| T-84-05 | Denial of Service | Rate limiting | accept | Vigil-core rate limits at 100 req/60s/IP. Extension sends at most 1 request per popup open. No amplification risk. |
</threat_model>
