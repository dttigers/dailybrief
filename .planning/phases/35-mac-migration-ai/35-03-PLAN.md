        // API-backed AI services
        triageService = APITriageService(client: client)
        insightService = APIInsightService(client: client)
        imageDescriptionService = APIImageDescriptionService(client: client)
        therapyClassificationService = APITherapyClassificationService(client: client)
        therapyPatternService = APITherapyPatternService(client: client)

        // ... etc (existing initialization unchanged)
