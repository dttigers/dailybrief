**Step A — Replace the stub at `Sources/vigil-watch/Commands/Run.swift`:**

```swift
// Phase 123 — Run subcommand body. Lifts Phase 122 main.swift composition root verbatim.
import ArgumentParser
import Foundation
import Darwin       // for dup2, open, O_WRONLY
import VigilWatch

struct Run: AsyncParsableCommand {
    static let configuration = CommandConfiguration(
        abstract: "Run daemon in foreground (default subcommand).",
        discussion: """
        Boots the FSEvents watcher, parses ~/.claude/projects/ JSONL, and POSTs to Vigil Core.
        Without --verbose, only NDJSON event lines hit stdout (pipe-friendly for `jq`).
        With --verbose, [INFO]/[WARN]/[ERROR] lifecycle logs also stream to stderr.
        """
    )

    @Flag(name: .customLong("verbose"),
          help: "Stream [INFO]/[WARN]/[ERROR] lifecycle logs to stderr.")
    var verbose: Bool = false

    func run() async throws {
        // CRITICAL: setbuf(stdout, nil) MUST be the first line — preserves Phase 122
        // pipe-friendly NDJSON behavior under launchd's stdout-redirection (RESEARCH §Pattern 1
        // "stdout buffering trap" lines 450-452). Plist-redirected stdout is fully buffered
        // by default in libc; this overrides that.
        setbuf(stdout, nil)

        // D-Discretion: without --verbose, suppress stderr lifecycle logs.
        // Phase 122 logInfo/logWarn/logError write to FileHandle.standardError, so the
        // cleanest path is to dup2 stderr to /dev/null BEFORE any log calls fire.
        // Errors are still surfaced because we use FileHandle.standardOutput for the
        // exit(1) path's "startup failed" message via stdout — but lifecycle [INFO]
        // chatter is silenced.
        if !verbose {
            let devnull = open("/dev/null", O_WRONLY)
            if devnull >= 0 {
                _ = dup2(devnull, fileno(stderr))
                close(devnull)
            }
        }

        // Phase 122 main.swift body — verbatim relocation:
        logInfo("vigil-watch starting (Phase 123)")
        do {
            let config = try ConfigLoader.load()
            let daemon = try await Daemon(config: config)
            installSIGTERMHandler(emitter: daemon.emitter)
            await daemon.start()
            logInfo("vigil-watch running — Ctrl-C to exit")
        } catch {
            // Errors during startup are fatal — surface to stderr REGARDLESS of --verbose.
            // (verbose-suppression dup2 only happened if !verbose; if we hit this path with
            // !verbose, stderr is /dev/null. Re-open it.)
            if !verbose {
                let dn = open("/dev/tty", O_WRONLY)
                if dn >= 0 { _ = dup2(dn, fileno(stderr)); close(dn) }
            }
            FileHandle.standardError.write(Data("vigil-watch startup failed: \(error)\n".utf8))
            throw ExitCode.failure
        }

        // FSEventStream + DispatchSource handlers require a runloop on the main thread.
        // RunLoop.main.run() does NOT return; SIGTERM/SIGINT handler calls exit(0).
        RunLoop.main.run()
    }
}
```

NOTES on the dup2 logic:
- `Darwin` is the macOS POSIX module (provides `open`, `dup2`, `close`, `fileno`).
- Without `--verbose`, stderr is replaced with `/dev/null`'s fd, so logInfo/logWarn/logError silently no-op.
- On startup-failure path, we attempt `/dev/tty` to surface the error; if that fails (e.g., no controlling tty under launchd), the error is lost — but launchd's `StandardErrorPath` will capture it via the daemon's pre-redirection write, and the `throw ExitCode.failure` exit code is still surfaced to launchd (which sees it as a failed-launch, KeepAlive then triggers a relaunch).

ALTERNATIVE: simpler approach. If the dup2 dance feels brittle, replace it with a wrapper sentinel that checks `verbose` before each log call. The trade-off: that requires changing every Phase 122 `logInfo`/`logWarn` call site downstream, which is much larger scope. Stick with dup2 — it's contained to Run.swift, no downstream impact.

```swift
import XCTest
import ArgumentParser
@testable import vigil_watch  // executable target name from Package.swift

final class RunSubcommandTests: XCTestCase {

    /// Without args, --verbose flag defaults to false.
    func testVerboseFlagDefaultIsFalse() throws {
        let cmd = try Run.parse([])
        XCTAssertFalse(cmd.verbose,
                       "--verbose default must be false (D-Discretion: 'Without --verbose, only NDJSON events go to stdout, stderr suppressed except errors').")
    }

    /// --verbose flag flips the bool.
    func testVerboseFlagEnabledByLongOption() throws {
        let cmd = try Run.parse(["--verbose"])
        XCTAssertTrue(cmd.verbose,
                      "--verbose long-form must enable the verbose lifecycle-log stream.")
    }

    /// CommandConfiguration declares the abstract that surfaces in --help.
    func testCommandHasNonEmptyAbstract() throws {
        XCTAssertFalse(Run.configuration.abstract.isEmpty,
                       "Run subcommand must have an abstract for --help discoverability.")
    }

    /// VALIDATION row 54 / Issue-2 BLOCKER: pin the actual stderr-suppression behavior of
    /// `Run.run()`'s dup2 dance — NOT just the parser flag default. The earlier 3 cases prove
    /// the flag wires through ArgumentParser; this case proves the side effect (stderr lifecycle
    /// log suppression when --verbose is OFF) is real, by capturing stderr around a full
    /// invocation of the dup2 + logInfo sequence.
    ///
    /// Strategy: run the body of Run.run()'s dup2-stderr block + a logInfo call inside a child
    /// scope where we have temporarily redirected fileno(stderr) to a pipe so we can capture
    /// the bytes that DID make it through. Then we restore the original stderr fd and assert.
    ///
    /// We DON'T invoke `Run().run()` directly because that calls `RunLoop.main.run()` which
    /// never returns. Instead we exercise the same dup2 sequence Run.run() uses, plus a
    /// logInfo() call that Run.run() makes ("vigil-watch starting (Phase 123)"), and assert
    /// the lifecycle log is present in capture WITH --verbose and absent WITHOUT.
    func testVerboseFlagEnablesStderrLogs() throws {
        // Helper: runs a closure with stderr captured to a Pipe; returns whatever bytes hit
        // FileHandle.standardError during the closure's execution.
        func captureStderr(_ block: () -> Void) -> String {
            // Save original fd 2 so we can restore it.
            let originalStderrFD = dup(fileno(stderr))
            defer { _ = dup2(originalStderrFD, fileno(stderr)); close(originalStderrFD) }

            let pipe = Pipe()
            _ = dup2(pipe.fileHandleForWriting.fileDescriptor, fileno(stderr))

            block()

            // Force-flush stderr buffers, then close the writing end so reading terminates.
            fflush(stderr)
            try? pipe.fileHandleForWriting.close()
            let data = pipe.fileHandleForReading.readDataToEndOfFile()
            return String(data: data, encoding: .utf8) ?? ""
        }

        // Helper: this is the EXACT dup2-then-log sequence Run.run() executes. We run it in
        // an isolated process-fd scope so each case is independent.
        func runStartupLogsSegment(verbose: Bool) {
            // Mirror Run.run()'s body up to (but not including) RunLoop.main.run():
            if !verbose {
                let devnull = open("/dev/null", O_WRONLY)
                if devnull >= 0 {
                    _ = dup2(devnull, fileno(stderr))
                    close(devnull)
                }
            }
            // Phase 122 logInfo writes to FileHandle.standardError. If --verbose suppressed
            // it via dup2 to /dev/null, the bytes are lost. If --verbose preserved stderr,
            // the bytes hit the captureStderr Pipe.
            logInfo("vigil-watch starting (Phase 123)")
            // Force any buffered output to drain before we restore stderr in captureStderr.
            fflush(stderr)
        }

        let withoutVerbose = captureStderr { runStartupLogsSegment(verbose: false) }
        XCTAssertFalse(withoutVerbose.contains("[INFO]"),
                       "Without --verbose, lifecycle [INFO] logs MUST be suppressed (D-Discretion). " +
                       "Captured stderr was: \(withoutVerbose)")
        XCTAssertFalse(withoutVerbose.contains("vigil-watch starting"),
                       "Without --verbose, the 'vigil-watch starting' lifecycle line MUST be suppressed.")

        let withVerbose = captureStderr { runStartupLogsSegment(verbose: true) }
        XCTAssertTrue(withVerbose.contains("[INFO]") || withVerbose.contains("vigil-watch starting"),
                      "With --verbose, lifecycle logs MUST stream to stderr. Got: \(withVerbose)")
    }
}
```

```swift
        "/usr/local/bin/jq",    // Intel Homebrew
        "/opt/homebrew/bin/jq", // Apple Silicon Homebrew
    ]
    for c in candidates where FileManager.default.isExecutableFile(atPath: c) {
        return c
    }
    return nil

/// Pure-function filter: applies the same `select(.session_id == $sid)` predicate as `jq`.
/// Used by tests that don't want to spawn subprocesses; kept in sync with jq's behavior.
/// Lines that aren't valid JSON are silently dropped (mirrors jq -c default).
internal func filterNDJSON(_ input: String, sessionId: String) -> [String] {
    var out: [String] = []
    for line in input.split(omittingEmptySubsequences: true, whereSeparator: { $0 == "\n" }) {
        guard let data = line.data(using: .utf8),
              let obj = try? JSONSerialization.jsonObject(with: data) as? [String: Any],
              let sid = obj["session_id"] as? String,
              sid == sessionId else { continue }
        out.append(String(line))
    }
    return out
}

struct Tail: ParsableCommand {
    static let configuration = CommandConfiguration(
        abstract: "Stream parsed events for one session via tail -f | jq.",
        discussion: """
        Filters ~/Library/Logs/Vigil/watch.out (Phase 122 NDJSON stdout, redirected by launchd
        in Plan 04) by session_id. Hangs until a matching event arrives — Ctrl-C to exit.
        Requires `jq` (system /usr/bin/jq on macOS 15+, brew jq otherwise).
        """
    )

    @Argument(help: "Session ID to filter on (matches NDJSON `session_id` field exactly).")
    var sessionId: String

    func run() throws {
        // 1. jq presence check (D-07) — fail loudly if absent.
        guard let jq = findJQPath() else {
            FileHandle.standardError.write(Data(
                "vigil-watch tail: jq not found; install with `brew install jq`\n".utf8
            ))
            throw ExitCode.failure
        }

        // 2. Build the pipeline.
        let logPath = NSHomeDirectory() + "/Library/Logs/Vigil/watch.out"
        let tail = Process()
        tail.executableURL = URL(fileURLWithPath: "/usr/bin/tail")
        tail.arguments = ["-f", logPath]

        let jqProc = Process()
        jqProc.executableURL = URL(fileURLWithPath: jq)
        // -c: compact output. --arg sid: pass session id as a jq variable.
        jqProc.arguments = ["-c", "--arg", "sid", sessionId, "select(.session_id == $sid)"]

        let pipe = Pipe()
        tail.standardOutput = pipe
        jqProc.standardInput = pipe
        jqProc.standardOutput = FileHandle.standardOutput
        jqProc.standardError = FileHandle.standardError

        // 3. SIGINT forwarding (T-123-04 mitigation) — lifted SHAPE from
        // Sources/VigilWatch/SignalHandling.swift:43-55. Hold the source alive in a local.
        let sigintSrc = DispatchSource.makeSignalSource(signal: SIGINT, queue: .main)
        sigintSrc.setEventHandler {
            tail.terminate()
            jqProc.terminate()
            exit(130)   // standard SIGINT exit code
        }
        signal(SIGINT, SIG_IGN)
        sigintSrc.resume()

        // 4. Run.
        do {
            try tail.run()
            try jqProc.run()
        } catch {
            FileHandle.standardError.write(Data("vigil-watch tail: failed to start pipeline: \(error)\n".utf8))
            throw ExitCode.failure
        }

        // jq exits when its stdin closes (i.e., when `tail` is terminated). We block on jq;
        // tail.terminate() runs after, then we exit.
        jqProc.waitUntilExit()
        tail.terminate()

        // Reference sigintSrc to keep the compiler from optimizing it out before this line.
        _ = sigintSrc
    }
}
```
