// Phase 123 — plist templates + shared subprocess helper.
// Both daemon and sampler launch agents live here as multi-line string templates

import Foundation

// MARK: - Plist templates

internal struct PlistTemplates {
    /// Daemon LaunchAgent template. Placeholders:
    /// - %BINARY_PATH%      : absolute path to /Users/<user>/.local/bin/vigil-watch
    /// - %LOG_DIR%          : absolute path to ~/Library/Logs/Vigil
    /// - %VIGIL_API_KEY_BLOCK% : either empty string (env var unset at install time) or
    ///                            "<key>VIGIL_API_KEY</key><string>vk_...</string>"
    ///
    /// IMPORTANT (Pitfall 2): booleans MUST be self-closing `<true/>` / `<false/>`.
    /// `<true></true>` validates with plutil but launchd's stricter parser rejects it.
    ///
    /// IMPORTANT (ROADMAP SC #1): KeepAlive is `<true/>` (boolean), NOT a `<dict>` like
    /// DailyBriefMonitor uses. This means launchd RESTARTS on any exit including clean ones —
    /// the soak gate (Plan 05) detects accidental KeepAlive=<dict> drift via "single PID across
    /// the 24h window" assertion.
    static let daemon: String = """
    <?xml version="1.0" encoding="UTF-8"?>
    <!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
    <plist version="1.0">
    <dict>
        <key>Label</key>
        <string>com.morrillholdings.vigil.watch</string>
        <key>ProgramArguments</key>
        <array>
            <string>%BINARY_PATH%</string>
            <string>run</string>
        </array>
        <key>RunAtLoad</key><true/>
        <key>KeepAlive</key><true/>
        <key>LimitLoadToSessionType</key><string>Aqua</string>
        <key>ProcessType</key><string>Interactive</string>
        <key>StandardOutPath</key><string>%LOG_DIR%/watch.out</string>
        <key>StandardErrorPath</key><string>%LOG_DIR%/watch.err</string>
        <key>EnvironmentVariables</key>
        <dict>
            <key>PATH</key>
            <string>/usr/local/bin:/usr/bin:/bin:/usr/sbin:/sbin</string>
            %VIGIL_API_KEY_BLOCK%
        </dict>
    </dict>
    </plist>
    """

    /// Sampler LaunchAgent — runs every 5 minutes (D-08).
    /// Placeholders:
    /// - %BINARY_PATH% : same as daemon (used in pgrep regex to discover the daemon PID).
    /// - %LOG_DIR%     : same as daemon — soak CSV path is %LOG_DIR%/soak-YYYY-MM-DD.csv.
    /// - %HOME%        : operator's home directory (e.g., /Users/jamesonmorrill).
    ///
    /// CRITICAL (Pitfall 4): use `etimes` (lowercase 's') NOT `etime`.
    /// - `etime` format is `[[dd-]hh:]mm:ss` — breaks awk parsers after 24h uptime.
    /// - `etimes` is integer seconds — supported on macOS 10.5+, stable forever.
    ///
    /// (Pitfall 5 mitigation): we use `pgrep -f <binary path>` here for simplicity. If a future
    /// soak run shows multi-PID ambiguity from concurrent `vigil-watch test` invocations,
    /// switch to `launchctl print gui/$UID/com.morrillholdings.vigil.watch | awk '/pid =/ ...'`.
    /// Plan 05's PID-uniqueness assertion will detect such drift.
    static let sampler: String = """
    <?xml version="1.0" encoding="UTF-8"?>
    <!DOCTYPE plist PUBLIC "-//Apple//DTD PLIST 1.0//EN" "http://www.apple.com/DTDs/PropertyList-1.0.dtd">
    <plist version="1.0">
    <dict>
        <key>Label</key>
        <string>com.morrillholdings.vigil.watch.sampler</string>
        <key>ProgramArguments</key>
        <array>
            <string>/bin/sh</string>
            <string>-c</string>
            <string>PID=$(pgrep -f %BINARY_PATH% | head -1); LOG=%LOG_DIR%/soak-$(date -u +%Y-%m-%d).csv; if [ -n "$PID" ]; then ps -p $PID -o pid=,rss=,etimes= | awk -v ts="$(date -u +%Y-%m-%dT%H:%M:%SZ)" '{print ts","$1","$2","$3}' >> "$LOG"; else echo "$(date -u +%Y-%m-%dT%H:%M:%SZ),,," >> "$LOG"; fi</string>
        </array>
        <key>RunAtLoad</key><true/>
        <key>StartInterval</key><integer>300</integer>
        <key>LimitLoadToSessionType</key><string>Aqua</string>
        <key>EnvironmentVariables</key>
        <dict>
            <key>PATH</key>
            <string>/usr/local/bin:/usr/bin:/bin:/usr/sbin:/sbin</string>
            <key>HOME</key>
            <string>%HOME%</string>
        </dict>
    </dict>
    </plist>
    """
}

// MARK: - XML escaping (T-123-03 / V5 Input Validation)

/// XML-escape a string for safe inclusion in a plist `<string>` element.
/// vk_ keys are alphanumeric so the practical risk of `&`/`<` is nil, but escape anyway
/// for correctness — a future bearer format that includes special chars would otherwise
/// silently produce malformed plist XML that bootstrap rejects.
internal func xmlEscape(_ s: String) -> String {
    var out = ""
    out.reserveCapacity(s.count)
    for ch in s {
        switch ch {
        case "&":  out += "&amp;"
        case "<":  out += "&lt;"
        case ">":  out += "&gt;"
        case "\"": out += "&quot;"
        case "'":  out += "&apos;"
        default:   out.append(ch)
        }
    }
    return out
}

// MARK: - Shared subprocess helper

internal struct ProcessResult {
    let exitCode: Int32
    let stdout: String
    let stderr: String
}

/// Run an external process and capture its exit code + stdout + stderr.
/// Used by Install (launchctl, plutil, which), Uninstall (launchctl), Status (launchctl print).
/// Lifted from RESEARCH.md §"Install subcommand skeleton" lines 700-711.
internal func runProcess(_ exe: String, _ args: [String]) throws -> ProcessResult {
    let p = Process()
    p.executableURL = URL(fileURLWithPath: exe)
    p.arguments = args
    let outPipe = Pipe()
    let errPipe = Pipe()
    p.standardOutput = outPipe
    p.standardError = errPipe
    try p.run()
    p.waitUntilExit()
    let out = String(data: outPipe.fileHandleForReading.readDataToEndOfFile(), encoding: .utf8) ?? ""
    let err = String(data: errPipe.fileHandleForReading.readDataToEndOfFile(), encoding: .utf8) ?? ""
    return ProcessResult(exitCode: p.terminationStatus, stdout: out, stderr: err)
}

        discussion: """
        Removes both LaunchAgent plists and unloads them from launchd. Logs and operator
        data under ~/Library/Application Support/vigil-watch/ are preserved (offsets,
        milestones, runtime-state — re-install picks up where this left off).

        let home = NSHomeDirectory()

        let samplerLabel = "com.morrillholdings.vigil.watch.sampler"
        let daemonPlist = "\(agentsDir)/\(daemonLabel).plist"
        let samplerPlist = "\(agentsDir)/\(samplerLabel).plist"

        let uid = String(getuid())

        // 1. Bootout both labels — tolerate exit 0 (was loaded) AND exit 3 (was not loaded).
        for label in [daemonLabel, samplerLabel] {
            let r = try runProcess("/bin/launchctl", ["bootout", "gui/\(uid)/\(label)"])
            if r.exitCode != 0 && r.exitCode != 3 {
                // Unexpected error (e.g. exit 5 I/O, exit 36 in progress) — log but continue

                FileHandle.standardError.write(Data(
                    "warning: launchctl bootout \(label) returned exit \(r.exitCode); continuing\n".utf8
                ))
            }
        }

        // 2. Remove both plists — tolerate ENOENT.
        for plist in [daemonPlist, samplerPlist] {
            if FileManager.default.fileExists(atPath: plist) {
                do {
                    try FileManager.default.removeItem(atPath: plist)
                    print("removed \(plist)")
                } catch {
                    FileHandle.standardError.write(Data(
                        "warning: failed to remove \(plist): \(error); continuing\n".utf8
                    ))
                }
            }
        }

        // 3. Operator data is INTENTIONALLY preserved per D-Discretion:
        //   - ~/Library/Application Support/vigil-watch/ (offsets.json, runtime-state.json, milestones)
        //   - ~/Library/Logs/Vigil/ (watch.{out,err}, soak-*.csv)
        //   - ~/.config/vigil/watch.toml
        // If the operator wants to wipe state, they can `rm -rf` those paths manually.

        print("vigil-watch uninstalled. Operator data preserved.")
        // Always exit 0 — uninstall is idempotent + best-effort.
    }
}

// Phase 123 — Status subcommand body.
// Reads runtime-state.json (from Plan 02 RuntimeStateWriter).
// Three-state output: RUNNING (fresh < 5s) / NOT RUNNING (stale or missing, but loaded) / NOT INSTALLED.
import ArgumentParser
import Foundation
import VigilWatch    // for RuntimeState type

/// Path injection seam for testing. Production uses RuntimeStateWriter.defaultPath.
internal struct StatusPaths {
    var runtimeStatePath: String
    var launchctlExecutable: String
    var daemonLabel: String

    static let `default` = StatusPaths(
        runtimeStatePath: RuntimeStateWriter.defaultPath,
        launchctlExecutable: "/bin/launchctl",
        daemonLabel: "com.morrillholdings.vigil.watch"
    )
}

struct Status: ParsableCommand {
    static let configuration = CommandConfiguration(
        abstract: "Print daemon state, queue depth, last event."
    )

    /// Test seam: tests set this before calling readStatus(); production uses the default.
    nonisolated(unsafe) static var injectedPaths: StatusPaths?

    /// Three-state outcome for unit testing.
    internal enum Outcome: Equatable {
        case running(RuntimeState)
        case notRunning(launchctlOutput: String)
        case notInstalled
    }

    /// Pure resolver: file-mtime + JSON read + launchctl-print fallback. Unit-testable.
    internal static func resolve(now: Date = Date(),
                                 paths: StatusPaths = StatusPaths.default) -> Outcome {
        let url = URL(fileURLWithPath: paths.runtimeStatePath)
        let fm = FileManager.default

        if let attrs = try? fm.attributesOfItem(atPath: paths.runtimeStatePath),
           let mtime = attrs[.modificationDate] as? Date,
           now.timeIntervalSince(mtime) < 5.0,
           let data = try? Data(contentsOf: url),
           let state = try? JSONDecoder().decode(RuntimeState.self, from: data) {
            return .running(state)
        }

        // File missing or stale → check whether launchd has the agent loaded.
        let uid = String(getuid())
        // (Note: in tests, runProcess invocation against a non-existent /bin/launchctl path
        // would throw — but on macOS hosts /bin/launchctl always exists, and tests use an
        // injected stub via injectedPaths. Belt-and-suspenders: catch.)
        let r: ProcessResult
        do {
            r = try runProcess(paths.launchctlExecutable,
                               ["print", "gui/\(uid)/\(paths.daemonLabel)"])
        } catch {
            return .notInstalled
        }
        if r.exitCode != 0 {
            return .notInstalled
        }
        return .notRunning(launchctlOutput: r.stdout)
    }

    func run() throws {
        let paths = Status.injectedPaths ?? StatusPaths.default
        let outcome = Status.resolve(now: Date(), paths: paths)

        switch outcome {
        case .running(let state):
            print("daemon: RUNNING")
            print("pid: \(state.pid)")
            print("started_at: \(state.startedAt)")
            print("queue_depth: \(state.queueDepth)")
            if let ts = state.lastEventTs,
               let sid = state.lastEventSessionId,
               let evt = state.lastEventType {
                print("last_event: \(evt) on \(sid) at \(ts)")
            } else {
                print("last_event: (none yet)")
            }
            print("quarantined: \(state.quarantined)")
            // exit 0 (return)

        case .notRunning(let launchctlOutput):
            print("daemon: NOT RUNNING (runtime-state stale or missing)")
            print("---")
            print(launchctlOutput)
            throw ExitCode(2)   // distinct from NOT INSTALLED

        case .notInstalled:
            print("daemon: NOT INSTALLED")
            print("hint: run `vigil-watch install`")
            throw ExitCode.failure
        }
    }
}

            [.modificationDate: Date().addingTimeInterval(-secondsAgo)],
            ofItemAtPath: path
        )
    }

    /// Fresh runtime-state.json (mtime now) → daemon RUNNING.
    func testReadsAndPrintsState() throws {
        try writeState(makeState())
        let paths = StatusPaths(
            runtimeStatePath: statePath,
            launchctlExecutable: "/bin/launchctl",
            daemonLabel: "nonexistent.test.label"   // launchctl print will fail; doesn't matter — file is fresh
        )
        let outcome = Status.resolve(now: Date(), paths: paths)

        guard case .running(let read) = outcome else {
            XCTFail("Expected .running outcome; got \(outcome)")
            return
        }
        XCTAssertEqual(read, makeState(),
                       "Status's parsed RuntimeState must round-trip through writer→file→reader.")
    }

    /// Stale file (mtime > 5s ago) AND launchctl print succeeds → NOT RUNNING.
    /// (We can't easily stub launchctl from unit tests; on a fresh CI host without our daemon installed,
    /// `launchctl print gui/$UID/com.morrillholdings.vigil.watch` returns non-zero.
    /// So we use a paths variant that points launchctl at our test daemon label —
    /// likely never loaded → outcome is .notInstalled.
    /// To distinguish .notRunning vs .notInstalled cleanly, we write a stale file and verify
    /// resolve() bypasses the file-fresh branch.)
    func testStaleFileBranchesAwayFromRunning() throws {
        try writeState(makeState())
        try setMTime(statePath, secondsAgo: 10.0)   // 10s old > 5s threshold

        let paths = StatusPaths(
            runtimeStatePath: statePath,
            launchctlExecutable: "/bin/launchctl",
            daemonLabel: "com.morrillholdings.vigil.watch.never.loaded.test"
        )
        let outcome = Status.resolve(now: Date(), paths: paths)

        // The exact downstream branch (.notRunning vs .notInstalled) depends on whether
        // launchctl reports the test label as loaded or not. For a label that was never
        // loaded, it'll be .notInstalled (exit non-zero).
        switch outcome {
        case .running:
            XCTFail("Stale file (10s old) must NOT branch to .running.")
        case .notRunning, .notInstalled:
            // Either is acceptable; the branch-away-from-running is what we're pinning.
            break
        }
    }

    /// Missing file AND launchctl print non-zero (label never loaded) → NOT INSTALLED.
    func testFallbackWhenNotInstalled() {
        // statePath does NOT exist (we never wrote it)
        let paths = StatusPaths(
            runtimeStatePath: statePath,
            launchctlExecutable: "/bin/launchctl",
            daemonLabel: "com.morrillholdings.vigil.watch.absolutely.does.not.exist"
        )
        let outcome = Status.resolve(now: Date(), paths: paths)
        XCTAssertEqual(outcome, .notInstalled,
                       "Missing runtime-state.json + unloaded launchd label must yield .notInstalled.")
    }

    /// 5s freshness boundary — exactly 5s ago is NOT fresh; 4s ago IS fresh.
    func testFreshnessBoundary() throws {
        try writeState(makeState())
        try setMTime(statePath, secondsAgo: 4.0)
        let outcome4s = Status.resolve(now: Date(),
                                       paths: StatusPaths(runtimeStatePath: statePath,
                                                         launchctlExecutable: "/bin/launchctl",
                                                         daemonLabel: "n/a"))
        XCTAssertNotEqual(outcome4s, .notInstalled,
                          "4s-old file must be considered fresh.")
        if case .running = outcome4s {} else {
            XCTFail("Expected .running for 4s-old file; got \(outcome4s)")
        }
    }
}
