**Step A — Create `scripts/soak-check.sh`:**

```bash
#!/usr/bin/env bash
# Phase 123 — 24h soak gate (D-09).
# Exits 0 iff:
#   - max(rss_kb) < 30000 (≤30MB resident)
#   - (last_ts - first_ts) >= 86400 - 600 (≥23h50m of sampling)
#   - All non-empty PID rows share one PID value
#   - At least one non-empty row exists
#   - vigil-core /v1/agent-sessions returns >0 sessions (skipped with --no-core-check)
#
# CSV schema (produced by sampler launchd agent, D-08):
#   ts,pid,rss_kb,etime_seconds
# Empty pid/rss/etime mark sampling intervals where the daemon was not running.

set -euo pipefail

NO_CORE_CHECK=false
CSV=""

# Parse args (one positional + optional --no-core-check).
for arg in "$@"; do
    case "$arg" in
        --no-core-check) NO_CORE_CHECK=true ;;
        --help|-h)
            echo "usage: $0 [--no-core-check] [CSV_PATH]" >&2
            echo "  CSV_PATH defaults to ~/Library/Logs/Vigil/soak-\$(date -u +%Y-%m-%d).csv" >&2
            exit 0
            ;;
        --*)
            echo "unknown flag: $arg" >&2
            exit 2
            ;;
        *)
            CSV="$arg"
            ;;
    esac
done

CSV="${CSV:-$HOME/Library/Logs/Vigil/soak-$(date -u +%Y-%m-%d).csv}"
[ -f "$CSV" ] || { echo "soak CSV missing: $CSV" >&2; exit 1; }

# 1. ≥1 non-empty row (sampler ran at all).
NONEMPTY=$(awk -F, '$2 != ""' "$CSV" | wc -l | tr -d ' ')
[ "$NONEMPTY" -ge 1 ] || { echo "FAIL: no non-empty rows — daemon never sampled" >&2; exit 1; }

# 2. Aggregate metrics.
MAX_RSS=$(awk -F, 'NR>0 && $3 != "" { if ($3 > m) m=$3 } END { print m+0 }' "$CSV")
PID_COUNT=$(awk -F, '$2 != "" { print $2 }' "$CSV" | sort -u | wc -l | tr -d ' ')
FIRST_TS=$(awk -F, '$2 != "" {print $1; exit}' "$CSV")
LAST_TS=$(awk -F, '$2 != "" {ts=$1} END {print ts}' "$CSV")
FIRST_S=$(date -u -j -f "%Y-%m-%dT%H:%M:%SZ" "$FIRST_TS" +%s 2>/dev/null || echo 0)
LAST_S=$(date -u -j -f "%Y-%m-%dT%H:%M:%SZ" "$LAST_TS" +%s 2>/dev/null || echo 0)
SPAN=$((LAST_S - FIRST_S))

echo "max RSS:      ${MAX_RSS} KB"
echo "unique PIDs:  ${PID_COUNT}"
echo "uptime span:  ${SPAN}s ($((SPAN/3600))h $((SPAN%3600/60))m)"
echo "samples:      ${NONEMPTY}"

# 3. RSS gate (D-09 / ROADMAP SC #4: ≤30MB resident).
[ "$MAX_RSS" -lt 30000 ] || { echo "FAIL: max RSS ${MAX_RSS} >= 30000 KB" >&2; exit 1; }

# 4. Single-PID gate (KeepAlive integrity — Pitfall 7 detector for Phase 122 SIGSEGV flake).
[ "$PID_COUNT" -eq 1 ] || { echo "FAIL: ${PID_COUNT} distinct PIDs (KeepAlive should have held one)" >&2; exit 1; }

# 5. Span gate (≥23h50m).
[ "$SPAN" -ge 85800 ] || { echo "FAIL: span ${SPAN}s < 85800s (23h50m gate)" >&2; exit 1; }

# 6. Live Core check (skippable for unit tests).
if [ "$NO_CORE_CHECK" = "true" ]; then
    echo "(skipped Core readback per --no-core-check)"
else
    [ -n "${VIGIL_API_KEY:-}" ] || { echo "FAIL: VIGIL_API_KEY env required for Core readback (or pass --no-core-check)" >&2; exit 1; }
    CORE_RESP=$(curl -fsS -H "Authorization: Bearer ${VIGIL_API_KEY}" \
        "https://api.vigilhub.io/v1/agent-sessions" | jq '.data | length')
    [ "${CORE_RESP:-0}" -gt 0 ] || { echo "FAIL: Core returned 0 sessions" >&2; exit 1; }
    echo "Core sessions: ${CORE_RESP}"
fi

echo
echo "PHASE 123 SOAK GATE: PASSED"
```

- `$2 != ""` filters rows where pid column is non-empty.
- The previous version (RESEARCH skeleton line 822) had `LAST_TS=$(awk -F, 'END {print $1}' "$CSV")` which would pick up a possibly-empty trailing row; the version above uses `awk -F, '$2 != "" {ts=$1} END {print ts}'` which is the LAST non-empty row's ts. import XCTest
import Foundation

final class SoakCheckTests: XCTestCase {

    /// Resolve repo root by walking up from the test file location until Package.swift is found.
    private func repoRoot() -> String {
        var dir = URL(fileURLWithPath: #filePath).deletingLastPathComponent()
        for _ in 0..<6 {
            if FileManager.default.fileExists(atPath: dir.appendingPathComponent("Package.swift").path) {
                return dir.path
            }
            dir.deleteLastPathComponent()
        }
        return FileManager.default.currentDirectoryPath
    }

    private func scriptPath() throws -> String {
        let path = repoRoot() + "/scripts/soak-check.sh"
        guard FileManager.default.fileExists(atPath: path) else {
            throw XCTSkip("scripts/soak-check.sh missing at \(path); run Plan 123-05 Task 5.1 first.")
        }
        guard FileManager.default.isExecutableFile(atPath: path) else {
            throw XCTSkip("scripts/soak-check.sh not executable at \(path); run `chmod +x scripts/soak-check.sh`.")
        }
        return path
    }

    /// Resolve a fixture CSV from the test bundle's resources.
    private func fixturePath(_ name: String) throws -> String {
        guard let url = Bundle.module.url(forResource: name.replacingOccurrences(of: ".csv", with: ""),
                                          withExtension: "csv",
                                          subdirectory: "Fixtures") else {
            throw XCTSkip("Fixture missing in test bundle: Fixtures/\(name)")
        }
        return url.path
    }

    /// Run soak-check.sh against a fixture with --no-core-check; capture exit + stdout + stderr.
    private func runSoak(fixture: String) throws -> (exitCode: Int32, stdout: String, stderr: String) {
        let script = try scriptPath()
        let csv = try fixturePath(fixture)
        let p = Process()
        p.executableURL = URL(fileURLWithPath: "/bin/bash")
        p.arguments = [script, "--no-core-check", csv]
        let outPipe = Pipe(); let errPipe = Pipe()
        p.standardOutput = outPipe
        p.standardError = errPipe
        try p.run()
        p.waitUntilExit()
        let out = String(data: outPipe.fileHandleForReading.readDataToEndOfFile(), encoding: .utf8) ?? ""
        let err = String(data: errPipe.fileHandleForReading.readDataToEndOfFile(), encoding: .utf8) ?? ""
        return (p.terminationStatus, out, err)
    }

    // MARK: - Cases (RESEARCH §Validation Architecture rows 988-991)

    func testGoodSoakPasses() throws {
        let r = try runSoak(fixture: "soak-good.csv")
        XCTAssertEqual(r.exitCode, 0,
                       "Good CSV must exit 0. stdout:\n\(r.stdout)\nstderr:\n\(r.stderr)")
        XCTAssertTrue(r.stdout.contains("PHASE 123 SOAK GATE: PASSED"),
                      "Good CSV must produce success marker. stdout:\n\(r.stdout)")
    }

    func testRSSAboveThresholdFails() throws {
        let r = try runSoak(fixture: "soak-rss-too-high.csv")
        XCTAssertNotEqual(r.exitCode, 0,
                          "RSS-too-high CSV must exit non-zero. stdout:\n\(r.stdout)")
        XCTAssertTrue(r.stderr.contains("FAIL: max RSS"),
                      "RSS-too-high failure must surface specific error. stderr:\n\(r.stderr)")
    }

    func testMultiplePIDsFails() throws {
        let r = try runSoak(fixture: "soak-multi-pid.csv")
        XCTAssertNotEqual(r.exitCode, 0,
                          "Multi-PID CSV must exit non-zero (KeepAlive integrity gate).")
        XCTAssertTrue(r.stderr.contains("distinct PIDs"),
                      "Multi-PID failure must surface specific error. stderr:\n\(r.stderr)")
    }

    func testShortSpanFails() throws {
        let r = try runSoak(fixture: "soak-short-span.csv")
        XCTAssertNotEqual(r.exitCode, 0,
                          "Short-span CSV must exit non-zero (20h span vs 23h50m gate).")
        XCTAssertTrue(r.stderr.contains("23h50m") || r.stderr.contains("85800"),
                      "Short-span failure must surface span gate. stderr:\n\(r.stderr)")
    }

    func testEmptyCSVFails() throws {

        XCTAssertNotEqual(r.exitCode, 0,
                          "Empty (all-pid-blank) CSV must exit non-zero.")
        XCTAssertTrue(r.stderr.contains("no non-empty rows"),
                      "Empty failure must surface specific error. stderr:\n\(r.stderr)")
    }
}
