```swift
// Phase 123 — Daemon-to-CLI status IPC.
// 1Hz snapshot file at ~/Library/Application Support/vigil-watch/runtime-state.json.
// Atomic temp+F_FULLFSYNC+rename — same primitive as Phase 122 StateStore (D-05).
import Foundation

/// D-04 schema_version=1 for the runtime-state.json snapshot.
public struct RuntimeState: Codable, Sendable, Equatable {
    public var schemaVersion: Int
    public var pid: Int32
    public var startedAt: String              // ISO-8601 UTC, captured at Daemon.init
    public var queueDepth: Int
    public var lastEventTs: String?
    public var lastEventSessionId: String?
    public var lastEventType: String?
    public var quarantined: Bool

    public init(schemaVersion: Int = 1, pid: Int32, startedAt: String,
                queueDepth: Int,
                lastEventTs: String? = nil,
                lastEventSessionId: String? = nil,
                lastEventType: String? = nil,
                quarantined: Bool) {
        self.schemaVersion       = schemaVersion
        self.pid                 = pid
        self.startedAt           = startedAt
        self.queueDepth          = queueDepth
        self.lastEventTs         = lastEventTs
        self.lastEventSessionId  = lastEventSessionId
        self.lastEventType       = lastEventType
        self.quarantined         = quarantined
    }

    /// D-04 NORMATIVE: JSON keys must be EXACTLY these snake_case strings.
    /// Status subcommand (Plan 04) reads them by the same names.
    public enum CodingKeys: String, CodingKey {
        case schemaVersion        = "schema_version"
        case pid
        case startedAt            = "started_at"
        case queueDepth           = "queue_depth"
        case lastEventTs          = "last_event_ts"
        case lastEventSessionId   = "last_event_session_id"
        case lastEventType        = "last_event_type"
        case quarantined
    }
}

public enum RuntimeStateWriterError: Error {
    case writeFailed(String)
}

/// Writes RuntimeState to disk atomically. Single-owner actor (one writer per process).
public actor RuntimeStateWriter {
    public static let defaultPath: String = {
        NSHomeDirectory() + "/Library/Application Support/vigil-watch/runtime-state.json"
    }()

    private let path: URL

    public init(path: String = defaultPath) {
        self.path = URL(fileURLWithPath: path)
    }

    /// Encode + atomic temp+F_FULLFSYNC+rename. Best-effort durability.
    /// If parent dir is missing it is created on first write.
    public func write(_ state: RuntimeState) throws {
        let dir = path.deletingLastPathComponent()
        if !FileManager.default.fileExists(atPath: dir.path) {
            try FileManager.default.createDirectory(at: dir, withIntermediateDirectories: true)
        }
        let tmp = dir.appendingPathComponent(path.lastPathComponent + ".tmp")

        let encoder = JSONEncoder()
        encoder.outputFormatting = [.prettyPrinted, .sortedKeys]

        let data: Data
        do { data = try encoder.encode(state) }
        catch { throw RuntimeStateWriterError.writeFailed("encode: \(error)") }

        do { try data.write(to: tmp) }
        catch { throw RuntimeStateWriterError.writeFailed("write temp: \(error)") }

        // F_FULLFSYNC: macOS-specific stronger fsync (matches Phase 122 StateStore).
        let fd = open(tmp.path, O_RDONLY)
        if fd != -1 { _ = fcntl(fd, F_FULLFSYNC); close(fd) }

        do {
            if FileManager.default.fileExists(atPath: path.path) {
                try? FileManager.default.removeItem(at: path)
            }
            try FileManager.default.moveItem(at: tmp, to: path)
        } catch {
            throw RuntimeStateWriterError.writeFailed("rename: \(error)")
        }
    }
}
```
