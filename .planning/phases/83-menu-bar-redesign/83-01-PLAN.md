```typescript
import { Hono } from "hono";
import { db } from "../db/connection.js";
import { appSettings } from "../db/schema.js";
import { eq } from "drizzle-orm";

export interface PrintSchedule {
  hour: number;
  minute: number;
  enabled: boolean;
}

const DEFAULT_SCHEDULE: PrintSchedule = { hour: 6, minute: 0, enabled: true };
const SETTINGS_KEY = "print_schedule";

// DI interface for testing
export interface SettingsDeps {
  dbGetFn?: () => Promise<PrintSchedule | null>;
  dbUpsertFn?: (s: PrintSchedule) => Promise<void>;
}

export function createSettingsRouter(deps?: SettingsDeps): Hono {
  const router = new Hono();

  router.get("/settings/print-schedule", async (c) => {
    try {
      let schedule: PrintSchedule;
      if (deps?.dbGetFn) {
        schedule = (await deps.dbGetFn()) ?? DEFAULT_SCHEDULE;
      } else {
        if (!db) return c.json({ error: "database_unavailable" }, 503);
        const rows = await db
          .select({ value: appSettings.value })
          .from(appSettings)
          .where(eq(appSettings.key, SETTINGS_KEY))
          .limit(1);
        schedule = rows.length > 0
          ? (rows[0].value as PrintSchedule)
          : DEFAULT_SCHEDULE;
      }
      return c.json(schedule, 200);
    } catch (err) {
      console.error("[settings] GET error:", err instanceof Error ? err.message : String(err));
      return c.json({ error: "internal_error" }, 500);
    }
  });

  router.put("/settings/print-schedule", async (c) => {
    try {
      const body = await c.req.json<unknown>();
      if (
        typeof body !== "object" || body === null ||
        typeof (body as PrintSchedule).hour !== "number" ||
        typeof (body as PrintSchedule).minute !== "number" ||
        typeof (body as PrintSchedule).enabled !== "boolean" ||
        (body as PrintSchedule).hour < 0 || (body as PrintSchedule).hour > 23 ||
        (body as PrintSchedule).minute < 0 || (body as PrintSchedule).minute > 59
      ) {
        return c.json({ error: "invalid_input" }, 400);
      }
      const schedule = body as PrintSchedule;

      if (deps?.dbUpsertFn) {
        await deps.dbUpsertFn(schedule);
      } else {
        if (!db) return c.json({ error: "database_unavailable" }, 503);
        await db.insert(appSettings)
          .values({ key: SETTINGS_KEY, value: schedule })
          .onConflictDoUpdate({
            target: appSettings.key,
            set: { value: schedule, updatedAt: new Date() },
          });
      }
      return c.json({ ok: true }, 200);
    } catch (err) {
      console.error("[settings] PUT error:", err instanceof Error ? err.message : String(err));
      return c.json({ error: "internal_error" }, 500);
    }
  });

  return router;
}

export const settings = createSettingsRouter();
```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| PWA → API | Authenticated request from browser carrying Bearer token |
| Mac app → API | Authenticated request from Swift carrying Bearer token |
| Public internet → API | Any unauthenticated caller attempting to reach /v1/settings/* |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-83-01 | Elevation of Privilege | GET /v1/settings/print-schedule | mitigate | Route registered after bearerAuth middleware in index.ts — any request without a valid `Authorization: Bearer <key>` header returns 401 before reaching the handler |
| T-83-02 | Elevation of Privilege | PUT /v1/settings/print-schedule | mitigate | Same bearerAuth protection as T-83-01 |
| T-83-03 | Tampering | PUT request body — hour/minute out of range | mitigate | Explicit bounds check: hour 0-23, minute 0-59, enabled must be boolean. Returns 400 on any violation (per Pitfall 5 in RESEARCH.md) |
| T-83-04 | Denial of Service | Repeated PUT calls flooding app_settings upsert | accept | Global rate limiter (100 req/60s per IP) already applied in index.ts via `rateLimiter` middleware. Single-row key-value upsert has negligible DB cost. |
| T-83-05 | Information Disclosure | app_settings value column leaks sensitive data | accept | Only stores { hour, minute, enabled } — non-sensitive scheduling preferences. No PII, no secrets. |
| T-83-06 | Spoofing | Forged Bearer token | accept | Handled by existing `bearerAuth` middleware (bcrypt key hash comparison). No new attack surface introduced by this route. |
</threat_model>
