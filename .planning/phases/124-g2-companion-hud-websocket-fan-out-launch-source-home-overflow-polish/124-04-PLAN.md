    // Phase 124 Plan 04 — G2-POLISH-07 drift detector + smoke.
    // Locks the 4-line body invariant via SOURCE-CONTENT inspection (mirror
    // Phase 123 Plan 03's drift-detector pattern: fs.readFile + substring
    // assertion). Pure-content tests do not exercise the SDK, so they run
    // fast and offline.

    import { test } from "node:test";
    import assert from "node:assert/strict";
    import { readFileSync } from "node:fs";
    import { fileURLToPath } from "node:url";
    import { dirname, resolve } from "node:path";

    const __dirname = dirname(fileURLToPath(import.meta.url));
    const HOME_SRC = resolve(__dirname, "../home.ts");
    const src = readFileSync(HOME_SRC, "utf-8");

    test("G2-POLISH-07 drift: bodyContent array has exactly 4 entries", () => {
      // Find the bodyContent array literal. Match between `const bodyContent = [`
      // and the matching `].join`.
      const m = src.match(/const\s+bodyContent\s*=\s*\[([\s\S]*?)\]\.join/);
      assert.ok(m, "bodyContent array literal found");
      const body = m![1];
      // Count top-level array entries: split on commas at depth 0.
      // For this codebase's simple shape (template strings + plain literals
      // without nested arrays), splitting on `,` and trimming non-empty
      // entries is sufficient. Filter out empty trailing comma artifacts.
      const entries = body
        .split(/,(?![^`]*`)/) // ignore commas inside template strings (heuristic)
        .map(s => s.trim())
        .filter(s => s.length > 0 && s !== ""); // drop empty
      assert.equal(entries.length, 4, `bodyContent should have 4 entries; got ${entries.length}: ${JSON.stringify(entries)}`);
    });

    test("G2-POLISH-07 drift: home.ts does not reference DIVIDER", () => {
      // DIVIDER may legitimately appear in a comment; strip line-comments first.
      const noComments = src.replace(/\/\/.*$/gm, "").replace(/\/\*[\s\S]*?\*\//g, "");
      assert.equal(noComments.includes("DIVIDER"), false, "DIVIDER reference removed from home.ts");
    });

    test("G2-POLISH-07 drift: home.ts does not reference affirmation parameter", () => {
      // The Affirmation SCREEN still exists; the AFFIRMATION CONSTANT may exist
      // elsewhere. We only forbid `affirmation.affirmation`, `: VigilAffirmation`,
      // and the parameter name `affirmation` in this file.
      const noComments = src.replace(/\/\/.*$/gm, "").replace(/\/\*[\s\S]*?\*\//g, "");
      assert.equal(/affirmation\.affirmation/.test(noComments), false, "no affirmation.affirmation reference");
      assert.equal(/:\s*VigilAffirmation/.test(noComments), false, "VigilAffirmation type annotation removed");
    });

    test("G2-POLISH-07 drift: buildHomeScreen takes one parameter (summary)", () => {
      const noComments = src.replace(/\/\/.*$/gm, "").replace(/\/\*[\s\S]*?\*\//g, "");
      // Match the export signature
      const m = noComments.match(/export\s+function\s+buildHomeScreen\s*\(([^)]*)\)/);
      assert.ok(m, "buildHomeScreen export found");
      // Count comma-separated params at the top level of the param list.
      const params = m![1].split(",").map(s => s.trim()).filter(Boolean);
      assert.equal(params.length, 1, `buildHomeScreen should take 1 param; got ${params.length}: ${m![1]}`);
      assert.ok(/summary/.test(params[0]), `param should be summary; got: ${params[0]}`);
    });
