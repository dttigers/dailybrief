    ```
    // vigil-core/scripts/smoke-test-verify-email.ts
    // Phase 113 Plan 05 — live e2e smoke for AUTH-11 verify-email + resend.
    //
    // Strategy (Specific Ideas in 113-RESEARCH.md):
    //   1. Use the seed user (already allowlisted). Do NOT register a new
    //      user — would require expanding the allowlist for an ephemeral email.
    //   2. Directly INSERT an email_verify token row into password_reset_tokens
    //      for the seed user. Compute the raw token + sha256 hex hash exactly
    //      like the production code path.
    //   3. POST the raw token to /v1/auth/verify-email against the live API.
    //   4. Assert response 200 { ok: true }.
    //   5. Query DB: assert users.email_verified_at is now non-null AND within
    //      the last 5 seconds AND password_reset_tokens.used_at is non-null
    //      for the row we inserted.
    //   6. Re-POST the same token; assert 400 (single-use enforced).
    //   7. Cleanup: delete the inserted token row (leave email_verified_at
    //      alone — it's idempotent backfill semantics).
    //
    // Note: real-inbox delivery (SC#1) is verified by the HUMAN-UAT checklist,
    // not this script. Programmatic email parsing requires Resend webhook
    // ingestion (deferred — Phase 111 boundary).
    //
    // Run:
    //   cd vigil-core
    //   npm run smoke-test:verify-email
    //
    // Or against prod:
    //   VIGIL_API_BASE=https://api.vigilhub.io DATABASE_URL=$RAILWAY_DB_URL \
    //     npm run smoke-test:verify-email

    import * as crypto from "node:crypto";
    import postgres from "postgres";

    const API_BASE = process.env["VIGIL_API_BASE"] || "https://api.vigilhub.io";
    const DB_URL = process.env["DATABASE_URL"];
    const SEED_EMAIL =
      process.env["VIGIL_SEED_EMAIL"] || "jamesonmorrill1@gmail.com";

    function log(msg: string): void {
      console.log(`[smoke-test:verify-email] ${msg}`);
    }
    function die(msg: string): never {
      console.error(`[smoke-test:verify-email] FAIL: ${msg}`);
      process.exit(1);
    }

    async function main(): Promise<void> {
      if (!DB_URL) die("DATABASE_URL is required");
      const sql = postgres(DB_URL, { max: 1 });

      try {
        // Step 1: Resolve the seed user.
        const userRows = await sql<
          { id: number; email: string; email_verified_at: Date | null }[]
        >`SELECT id, email, email_verified_at FROM users WHERE email = ${SEED_EMAIL.toLowerCase()} LIMIT 1`;
        const user = userRows[0];
        if (!user) die(`seed user ${SEED_EMAIL} not found in DB at ${DB_URL}`);
        log(`seed user id=${user.id} email=${user.email} (current emailVerifiedAt=${user.email_verified_at?.toISOString() ?? "null"})`);

        // Step 2: Directly insert an email_verify token row.
        const rawToken = crypto.randomBytes(32).toString("base64url");
        const tokenHash = crypto
          .createHash("sha256")
          .update(rawToken)
          .digest("hex");
        const expiresAt = new Date(Date.now() + 24 * 60 * 60 * 1000);
        const insertRows = await sql<{ id: number }[]>`
          INSERT INTO password_reset_tokens
            (user_id, token_hash, type, expires_at)
          VALUES
            (${user.id}, ${tokenHash}, 'email_verify', ${expiresAt})
          RETURNING id
        `;
        const insertedTokenId = insertRows[0]?.id;
        if (!insertedTokenId) die("failed to INSERT token row");
        log(`inserted token row id=${insertedTokenId} (raw token len=${rawToken.length})`);

        // Step 3 + 4: POST to /v1/auth/verify-email; assert 200.
        const t0 = Date.now();
        const res = await fetch(`${API_BASE}/v1/auth/verify-email`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ token: rawToken }),
        });
        const elapsed = Date.now() - t0;
        const body = await res.json().catch(() => ({}));
        log(`POST /v1/auth/verify-email: status=${res.status} elapsed=${elapsed}ms body=${JSON.stringify(body)}`);
        if (res.status !== 200) die(`expected 200, got ${res.status}`);
        if ((body as { ok?: unknown }).ok !== true) {
          die(`expected body.ok=true, got ${JSON.stringify(body)}`);
        }

        // Step 5: Assert DB state.
        const userAfter = await sql<
          { email_verified_at: Date | null }[]
        >`SELECT email_verified_at FROM users WHERE id = ${user.id} LIMIT 1`;
        const newEmailVerifiedAt = userAfter[0]?.email_verified_at ?? null;
        if (!newEmailVerifiedAt) die("users.email_verified_at is still null after verify");
        const ageMs = Date.now() - newEmailVerifiedAt.getTime();
        if (ageMs > 10_000) {
          die(`email_verified_at=${newEmailVerifiedAt.toISOString()} is older than 10s — was not bumped by this verify call`);
        }
        log(`users.email_verified_at bumped to ${newEmailVerifiedAt.toISOString()} (age=${ageMs}ms — within 10s)`);

        const tokenAfter = await sql<{ used_at: Date | null }[]>`
          SELECT used_at FROM password_reset_tokens WHERE id = ${insertedTokenId} LIMIT 1
        `;
        if (!tokenAfter[0]?.used_at) die("token row used_at is still null after verify");
        log(`token row used_at=${tokenAfter[0].used_at.toISOString()} (single-use claim succeeded)`);

        // Step 6: Re-POST the same token; assert 400 (single-use).
        const res2 = await fetch(`${API_BASE}/v1/auth/verify-email`, {
          method: "POST",
          headers: { "Content-Type": "application/json" },
          body: JSON.stringify({ token: rawToken }),
        });
        const body2 = await res2.json().catch(() => ({}));
        log(`replay POST /v1/auth/verify-email: status=${res2.status} body=${JSON.stringify(body2)}`);
        if (res2.status !== 400) die(`expected 400 on replay, got ${res2.status}`);

        // Step 7: Cleanup — delete the inserted token row.
        await sql`DELETE FROM password_reset_tokens WHERE id = ${insertedTokenId}`;
        log(`cleanup: deleted token row id=${insertedTokenId}`);

        log("ALL CHECKS PASSED — AUTH-11 verify-email + single-use semantics live-verified");
      } finally {
        await sql.end({ timeout: 5 });
      }
    }

    main().catch((err) => {
      console.error("[smoke-test:verify-email] uncaught:", err);
      process.exit(1);
    });
    ```
