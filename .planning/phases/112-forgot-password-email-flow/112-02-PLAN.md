<objective>
Ship the unauthenticated `POST /v1/auth/forgot-password` endpoint that accepts
an email, ALWAYS responds 200 with the same enumeration-safe body (D-03), runs
a dummy argon2 verify on the miss path so wall-clock time approximates the hit
path (D-05, SC#1), and on the hit path generates a 256-bit base64url token
(D-07), stores its SHA-256 hex hash (D-08, SC#5), invalidates any prior unused
tokens for the user (D-06), and dispatches the password-reset email via Phase
111's `sendPasswordResetEmail` wrapper.

**Purpose:** This is the front door for AUTH-10. SC#1 (enumeration safety) is
the load-bearing security property — an attacker who can distinguish "this
email exists" vs "this email doesn't" defeats the privacy boundary. Two
independent mitigations stack: (a) identical response body and status (D-03),
and (b) approximate wall-clock parity via dummy argon2 verify (D-05). The dual
rate limit (5/hour per-IP AND per-email, RESEARCH §Pattern-3) prevents two
distinct abuse vectors: per-IP catches a script spamming arbitrary emails to
exhaust Resend quota, per-email catches a known-good email being weaponized as
a spam target.

```typescript
// vigil-core/src/routes/forgot-password.ts (NEW)
import { Hono } from "hono";
import { eq, and, isNull } from "drizzle-orm";
import * as crypto from "node:crypto";
import { db } from "../db/connection.js";
import { users, passwordResetTokens } from "../db/schema.js";
import { verifyPassword } from "../utils/password.js";
import { sendPasswordResetEmail } from "../services/email-service.js";

// Dummy argon2 hash for timing-safe forgot-password on unknown email.
// Keep in sync with vigil-core/src/routes/auth.ts:17 — same OPTIONS (m=19456,
// t=2, p=1, argon2id). RESEARCH §A2 verified params match utils/password.ts:OPTIONS.
const DUMMY_HASH =
  "$argon2id$v=19$m=19456,t=2,p=1$ZHVtbXlzYWx0ZHVtbXlzYWw$ZHVtbXloYXNoZHVtbXloYXNoZHVtbXloYXNoZHVtbXk";

const TOKEN_TTL_MS = 60 * 60 * 1000;             // D-21 1h expiry
const RATE_LIMIT_MAX = 5;                         // D-04
const RATE_LIMIT_WINDOW_MS = 60 * 60 * 1000;      // D-04 (1h sliding window)

const ENUM_SAFE_BODY = {
  ok: true,
  message: "If your account exists, a reset link has been sent.",
};

const ipBuckets = new Map<string, number[]>();
const emailBuckets = new Map<string, number[]>();

function takeSlot(map: Map<string, number[]>, key: string, now: number): boolean {
  const cutoff = now - RATE_LIMIT_WINDOW_MS;
  const arr = (map.get(key) ?? []).filter((t) => t >= cutoff);
  if (arr.length >= RATE_LIMIT_MAX) { map.set(key, arr); return false; }
  arr.push(now); map.set(key, arr); return true;
}

export const forgotPassword = new Hono();

forgotPassword.post("/auth/forgot-password", async (c) => {
  // Body parse — invalid JSON STILL returns enum-safe success (D-03 hides input shape)
  let body: unknown;
  try { body = await c.req.json(); } catch { return c.json(ENUM_SAFE_BODY, 200); }

  const rawEmail = (body as { email?: unknown })?.email;
  const email = typeof rawEmail === "string" ? rawEmail.toLowerCase().trim() : null;

  // Dual-axis rate limit. Per orchestrator-resolved Open Q §1: return 200 enum-safe
  // body even when limited (hides which axis fired AND that limiting occurred at all).
  const ip = c.req.header("x-forwarded-for")?.split(",")[0]?.trim() ?? "unknown";
  const now = Date.now();
  const ipOk = takeSlot(ipBuckets, ip, now);
  const emailOk = email ? takeSlot(emailBuckets, email, now) : true;
  if (!ipOk || !emailOk) return c.json(ENUM_SAFE_BODY, 200);

  if (!email) return c.json(ENUM_SAFE_BODY, 200);
  if (!db) return c.json({ error: "Database unavailable" }, 503);

  const [user] = await db.select({ id: users.id, email: users.email })
                         .from(users).where(eq(users.email, email)).limit(1);

  if (!user) {
    // D-05 timing-attack mitigation
    await verifyPassword("never-matches", DUMMY_HASH);
    return c.json(ENUM_SAFE_BODY, 200);
  }

  // D-06: invalidate prior unused tokens for (user, 'password_reset')
  await db.update(passwordResetTokens).set({ usedAt: new Date() })
          .where(and(
            eq(passwordResetTokens.userId, user.id),
            eq(passwordResetTokens.type, "password_reset"),
            isNull(passwordResetTokens.usedAt),
          ));

  // D-07/D-08: 32 random bytes base64url; SHA-256 hex stored
  const rawToken = crypto.randomBytes(32).toString("base64url");
  const tokenHash = crypto.createHash("sha256").update(rawToken).digest("hex");

  await db.insert(passwordResetTokens).values({
    userId: user.id,
    tokenHash,
    type: "password_reset",
    expiresAt: new Date(now + TOKEN_TTL_MS),
  });

  const origin = process.env["VIGIL_APP_BASE_URL"] || "https://app.vigilhub.io";
  const resetUrl = `${origin}/auth/reset?token=${rawToken}`;
  // Result observed (logged on failure inside email-service via captureException);
  // does NOT change the response (D-03 invariant).
  await sendPasswordResetEmail(user.email, resetUrl);

  return c.json(ENUM_SAFE_BODY, 200);
});
```

```typescript
// Keep in sync with vigil-core/src/routes/auth.ts:17 — same OPTIONS
// (Argon2id, m=19456, t=2, p=1). RESEARCH §A2 verified params match
// utils/password.ts:OPTIONS exactly. Plaintext "never-matches" verifies in
// ~100-200ms wall-clock (argon2 is deliberately slow); this dominates the
// miss-path budget so timing approximates the hit-path within ~30%.

  "$argon2id$v=19$m=19456,t=2,p=1$ZHVtbXlzYWx0ZHVtbXlzYWw$ZHVtbXloYXNoZHVtbXloYXNoZHVtbXloYXNoZHVtbXk";

const RATE_LIMIT_MAX = 5;                         // D-04
const RATE_LIMIT_WINDOW_MS = 60 * 60 * 1000;      // D-04: 1h sliding window

// ── Enumeration-safe response body (D-03 verbatim) ──────────────────────────
// Same body, same status, same wall-clock time on hit AND miss paths.
const ENUM_SAFE_BODY = {
  ok: true,
  message: "If your account exists, a reset link has been sent.",
} as const;

// ── In-process sliding-window rate limit buckets (RESEARCH §Pattern-3) ──────
// Per-IP and per-email tracked independently; "whichever fires first" wins.
// Single-instance scale only — fine for v3.6 Railway deployment. Multi-instance
// scale-out would need Redis (deferred to v3.7+).
const ipBuckets = new Map<string, number[]>();
const emailBuckets = new Map<string, number[]>();

// Periodic sweep — drop entries whose newest timestamp is outside the window.
// .unref() so the timer doesn't keep the process alive.
setInterval(() => {
  const cutoff = Date.now() - RATE_LIMIT_WINDOW_MS;
  for (const [k, arr] of ipBuckets) {
    if (arr.length === 0 || arr[arr.length - 1] < cutoff) ipBuckets.delete(k);
  }
  for (const [k, arr] of emailBuckets) {
    if (arr.length === 0 || arr[arr.length - 1] < cutoff) emailBuckets.delete(k);
  }
}, RATE_LIMIT_WINDOW_MS).unref();

function takeSlot(map: Map<string, number[]>, key: string, now: number): boolean {
  const cutoff = now - RATE_LIMIT_WINDOW_MS;
  const arr = (map.get(key) ?? []).filter((t) => t >= cutoff);
  if (arr.length >= RATE_LIMIT_MAX) {
    map.set(key, arr);
    return false;
  }
  arr.push(now);
  map.set(key, arr);
  return true;
}

export function createForgotPasswordRoute(deps?: ForgotPasswordDeps): Hono {
  const sendEmailFn = deps?.sendEmailFn ?? realSendPasswordResetEmail;
  const nowFn = deps?.nowFn ?? Date.now;
  const router = new Hono();

  router.post("/auth/forgot-password", async (c) => {
    // Body parse — invalid JSON → 200 enum-safe (no shape leak).
    let body: unknown;
    try {
      body = await c.req.json();
    } catch {
      return c.json(ENUM_SAFE_BODY, 200);
    }

    const rawEmail = (body as { email?: unknown })?.email;
    const email = typeof rawEmail === "string" ? rawEmail.toLowerCase().trim() : null;

    // Dual-axis rate limit — exceed either → 200 enum-safe (per orchestrator
    // resolution of RESEARCH §Open-Question-1: hide both which-axis-fired AND
    // that limiting occurred at all; 200 is preferred at single-user scale).
    const ip = c.req.header("x-forwarded-for")?.split(",")[0]?.trim() ?? "unknown";
    const now = nowFn();
    const ipOk = takeSlot(ipBuckets, ip, now);
    const emailOk = email ? takeSlot(emailBuckets, email, now) : true;
    if (!ipOk || !emailOk) {
      return c.json(ENUM_SAFE_BODY, 200);
    }

    // Missing email → 200 enum-safe (no shape leak)
    if (!email) {
      return c.json(ENUM_SAFE_BODY, 200);
    }

    // DB unavailable is an infrastructure failure, NOT an enumeration leak —
    // 503 is correct here (CONTEXT D-03 says "regardless of whether the email
    // maps to a user", not "regardless of DB availability").
    if (!db) {
      return c.json({ error: "Database unavailable" }, 503);
    }

    const [user] = await db
      .select({ id: users.id, email: users.email })
      .from(users)
      .where(eq(users.email, email))
      .limit(1);

    if (!user) {
      // D-05: timing-attack mitigation — burn ~100-200ms argon2 verify on miss.
      // Same OPTIONS (Argon2id, m=19456, t=2, p=1) as hashPassword in
      // utils/password.ts. RESEARCH §A2 confirms params identity.
      await verifyPassword("never-matches", DUMMY_HASH);
      return c.json(ENUM_SAFE_BODY, 200);
    }

    // (user_id, 'password_reset') BEFORE issuing a fresh one. Keeps single-use
    // semantics tight; user only has one valid link in inbox at a time.
    await db
      .update(passwordResetTokens)
      .set({ usedAt: new Date(now) })
      .where(and(
        eq(passwordResetTokens.userId, user.id),
        eq(passwordResetTokens.type, "password_reset"),
        isNull(passwordResetTokens.usedAt),
      ));

    // D-07: 32 random bytes (256 bits entropy) base64url-encoded.
    // D-08: SHA-256 hex of raw token is what gets stored (raw NEVER touches DB).
    const rawToken = crypto.randomBytes(32).toString("base64url");
    const tokenHash = crypto.createHash("sha256").update(rawToken).digest("hex");

    await db.insert(passwordResetTokens).values({
      userId: user.id,
      tokenHash,
      type: "password_reset",
      expiresAt: new Date(now + TOKEN_TTL_MS),
    });

    // D-21: VIGIL_APP_BASE_URL with prod fallback (matches Phase 111
    // smoke-test-email.ts:21 origin shape verbatim — `||` not `??` so

    const origin = process.env["VIGIL_APP_BASE_URL"] || "https://app.vigilhub.io";
    const resetUrl = `${origin}/auth/reset?token=${rawToken}`;

    // failure) but DOES NOT change response shape — D-03 invariant.
    await sendEmailFn(user.email, resetUrl);

    return c.json(ENUM_SAFE_BODY, 200);
  });

  return router;
}
```
