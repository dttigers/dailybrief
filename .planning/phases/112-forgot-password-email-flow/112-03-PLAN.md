```typescript
import { and, eq, isNull, gt, sql } from "drizzle-orm";
import * as crypto from "node:crypto";

const tokenHash = crypto.createHash("sha256").update(token).digest("hex");

const claimed = await db
  .update(passwordResetTokens)
  .set({ usedAt: new Date() })
  .where(and(
    eq(passwordResetTokens.tokenHash, tokenHash),
    eq(passwordResetTokens.type, "password_reset"),
    isNull(passwordResetTokens.usedAt),
    gt(passwordResetTokens.expiresAt, sql`now()`),
  ))
  .returning({ userId: passwordResetTokens.userId });

if (claimed.length === 0) {
  return c.json({ error: "Invalid or expired token" }, 400);
}

const newHash = await hashPassword(newPassword);
const ts = new Date();
await db
  .update(users)
  .set({ passwordHash: newHash, passwordChangedAt: ts, updatedAt: ts })
  .where(eq(users.id, claimed[0].userId));

return c.json({ ok: true, message: "Password reset successful. You can now log in." });
```

**Concurrency guarantee (RESEARCH §Pattern-2):** PostgreSQL acquires a row lock
on the matched row inside the UPDATE … WHERE — concurrent claims serialize, and
only the first sees `used_at IS NULL` true. The second's WHERE no longer
matches (used_at was just set), so it returns 0 rows. **No transaction wrapper
needed.**

**`gt(expiresAt, sql\`now()\`)` vs `gt(expiresAt, new Date())` choice
(RESEARCH §Pattern-2 last paragraph):** Use `sql\`now()\`` for byte-exact
symmetry with CONTEXT D-02's SQL spec and to use PG's clock not JS's. Drift
between PG clock and JS clock is sub-millisecond at single-instance scale, but
`sql\`now()\`` is the documented choice.

**State-mutation order (D-11 / orchestrator constraint #3):**
1. UPDATE password_reset_tokens claim (atomic single-use)
2. UPDATE users SET passwordHash + passwordChangedAt + updatedAt = now()
3. Return 200

If step 2 fails AFTER step 1 succeeded → token is already burned. Acceptable
failure mode per CONTEXT D-11 last paragraph; user requests fresh reset.

function sha256Hex(s: string): string {
  return crypto.createHash("sha256").update(s).digest("hex");

async function seedUser(email: string, password = "InitialPass123") {
  if (!db) throw new Error("DB unavailable");

  if (!db) throw new Error("DB unavailable");

    tokenHash,

    expiresAt: opts?.expiresAt ?? new Date(Date.now() + 60 * 60 * 1000),
    usedAt: opts?.usedAt ?? null,

  return { rawToken, tokenHash };

import { Hono } from "hono";
import { eq, and, isNull, gt, sql } from "drizzle-orm";
import * as crypto from "node:crypto";
import { db as defaultDb } from "../db/connection.js";
import { users, passwordResetTokens } from "../db/schema.js";

// ── Constants (keep in sync with vigil-core/src/routes/auth.ts:19-20) ──────
const MIN_PASSWORD = 12;
const MAX_PASSWORD = 128;

// ── Rate limit (D-13: per-IP only, 5/hour, sliding window) ─────────────────
const RATE_LIMIT_MAX = 5;
const RATE_LIMIT_WINDOW_MS = 60 * 60 * 1000;
const ipBuckets = new Map<string, number[]>();

setInterval(() => {
  const cutoff = Date.now() - RATE_LIMIT_WINDOW_MS;
  for (const [k, arr] of ipBuckets) {
    if (arr.length === 0 || arr[arr.length - 1] < cutoff) ipBuckets.delete(k);
  }
}, RATE_LIMIT_WINDOW_MS).unref();

function takeSlot(key: string, now: number): { ok: boolean; retryAfterSec: number } {
  const cutoff = now - RATE_LIMIT_WINDOW_MS;
  const arr = (ipBuckets.get(key) ?? []).filter((t) => t >= cutoff);
  if (arr.length >= RATE_LIMIT_MAX) {
    const retryAfterSec = Math.ceil((arr[0] + RATE_LIMIT_WINDOW_MS - now) / 1000);
    ipBuckets.set(key, arr);
    return { ok: false, retryAfterSec };
  }
  arr.push(now);
  ipBuckets.set(key, arr);
  return { ok: true, retryAfterSec: 0 };
}

export function __resetBucketsForTest(): void {
  ipBuckets.clear();
}

export interface ResetPasswordDeps {
  /** DI seam for tests — defaults to the real `db` from connection.js */
  dbOverride?: typeof defaultDb;
  /** DI seam for tests — defaults to Date.now */
  nowFn?: () => number;

export function createResetPasswordRoute(deps?: ResetPasswordDeps): Hono {
  const db = deps?.dbOverride ?? defaultDb;
  const nowFn = deps?.nowFn ?? Date.now;
  const router = new Hono();

  router.post("/auth/reset-password", async (c) => {
    // D-13: per-IP rate limit FIRST — token-guessing brute force is the threat.
    const ip = c.req.header("x-forwarded-for")?.split(",")[0]?.trim() ?? "unknown";
    const now = nowFn();
    const slot = takeSlot(ip, now);
    if (!slot.ok) {
      c.header("Retry-After", String(slot.retryAfterSec));
      return c.json({ error: "Too many requests" }, 429);
    }

    // Body parse
    let body: unknown;
    try {
      body = await c.req.json();
    } catch {
      return c.json({ error: "Invalid JSON body" }, 400);
    }
    const { token, newPassword } = (body ?? {}) as { token?: unknown; newPassword?: unknown };

    // Field guards
    if (typeof token !== "string" || typeof newPassword !== "string") {
      return c.json({ error: "token and newPassword are required" }, 400);
    }

    // Length validation BEFORE atomic claim — must NOT burn token on validation failure
    if (newPassword.length < MIN_PASSWORD || newPassword.length > MAX_PASSWORD) {
      return c.json(
        { error: `Password must be ${MIN_PASSWORD}-${MAX_PASSWORD} characters` },
        400,
      );
    }

    if (!db) return c.json({ error: "Database unavailable" }, 503);

    // D-10: atomic single-use claim via UPDATE-RETURNING. RESEARCH §Pattern-2.
    // PostgreSQL row-lock makes the WHERE+SET atomic; concurrent requests
    // serialize and only the first sees used_at IS NULL true. No transaction
    // needed.
    //
    // gt(expiresAt, sql`now()`) uses PG clock for byte-exact symmetry with
    // CONTEXT D-02's spec.
    const tokenHash = crypto.createHash("sha256").update(token).digest("hex");
    const claimed = await db
      .update(passwordResetTokens)
      .set({ usedAt: new Date(now) })
      .where(and(
        eq(passwordResetTokens.tokenHash, tokenHash),
        eq(passwordResetTokens.type, "password_reset"),
        isNull(passwordResetTokens.usedAt),
        gt(passwordResetTokens.expiresAt, sql`now()`),
      ))
      .returning({ userId: passwordResetTokens.userId });

    if (claimed.length === 0) {
      // Single-bucket: invalid OR expired OR used. D-20 (PWA single-bucket UX)
      // mirrored on the server — prevents info leak, keeps state machine simple.
      return c.json({ error: "Invalid or expired token" }, 400);
    }

    // D-11 step 2: update password + bump password_changed_at + updatedAt.
    // Phase 110 D-12: passwordChangedAt bump invalidates pre-reset JWTs via the
    // bearerAuth iat-gate (vigil-core/src/middleware/auth.ts:110-145). No new
    // gate code in this plan — the gate fires automatically on next request.
    //
    // CONTEXT D-11 last paragraph: if this UPDATE throws AFTER the claim
    // succeeded above, the token is already burned. User requests a fresh
    // reset. Acceptable failure mode — tested via mock DB injection.
    const newHash = await hashPassword(newPassword);
    const ts = new Date(now);
    await db
      .update(users)
      .set({ passwordHash: newHash, passwordChangedAt: ts, updatedAt: ts })
      .where(eq(users.id, claimed[0].userId));

    // D-12: NO JWT, NO auto-login, NO token in response. PWA navigates to
    // /auth?reason=password_reset on this 200 (Plan 04).
    return c.json({ ok: true, message: "Password reset successful. You can now log in." });
  });

  return router;
}
