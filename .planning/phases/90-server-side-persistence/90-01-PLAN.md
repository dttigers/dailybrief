2. Add GET /insights/cache endpoint BEFORE the POST handler:

```typescript
insights.get("/insights/cache", async (c) => {
  if (!db) return c.json({ error: "Database not available" }, 503);

  const rows = await db
    .select()
    .from(aiCache)
    .where(eq(aiCache.type, "insights"))
    .limit(1);

  if (rows.length === 0) {
    return c.json({ cached: false }, 404);
  }

  return c.json({
    insights: rows[0].result,
    cached: true,
    generatedAt: rows[0].generatedAt.toISOString(),
  });
});
```

2. Add GET /therapy/cache endpoint BEFORE the POST handlers:

```typescript
therapy.get("/therapy/cache", async (c) => {
  if (!db) return c.json({ error: "Database not available" }, 503);

  const type = c.req.query("type");
  if (type !== "patterns" && type !== "prep") {
    return c.json({ error: "type must be 'patterns' or 'prep'" }, 400);
  }

  const rows = await db
    .select()
    .from(aiCache)
    .where(eq(aiCache.type, type))
    .limit(1);

  if (rows.length === 0) {
    return c.json({ cached: false }, 404);
  }

  return c.json({
    ...(type === "patterns" ? { patterns: rows[0].result } : rows[0].result),
    cached: true,
    generatedAt: rows[0].generatedAt.toISOString(),
  });
});
```

For `type=patterns`, the response shape is `{ patterns: [...], cached: true, generatedAt: "..." }`.
For `type=prep`, the response shape is `{ items: [...], overallThemes: [...], suggestedFocus: "...", cached: true, generatedAt: "..." }` (spread the TherapyPrep object).
