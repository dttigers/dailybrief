    d) Replace the 429 branch in handleSubmit (currently lines 67-70). The full handleSubmit AFTER edit:
    ```typescript
    async function handleSubmit(e: React.FormEvent) {
      e.preventDefault()
      if (!token) {
        setTokenInvalid(true)
        return
      }
      setSubmitting(true)
      setError(null)
      try {
        const res = await fetch(`${API_BASE}/v1/auth/reset-password`, {
          method: 'POST',
          headers: { 'Content-Type': 'application/json' },
          body: JSON.stringify({ token, newPassword: newPw }),
        })
        if (res.status === 200) {
          navigate('/auth?reason=password_reset')
          return
        }
        if (res.status === 400) {
          // D-20: invalid OR expired OR used → single-bucket UX.
          setTokenInvalid(true)
          return
        }
        if (res.status === 429) {
          // Phase 117 (AUTH-12 D-10/D-11): rate-limited bucket with countdown.
          const errorClass = await classifyFetchError(res)
          if (errorClass.kind === 'rate-limited') {
            setRateLimited(true)
            if (errorClass.retryAfter !== undefined) {
              const seconds = errorClass.retryAfter
              setRetryCountdown(seconds)
              if (countdownTimerRef.current !== null) {
                window.clearInterval(countdownTimerRef.current)
              }
              const timerId = window.setInterval(() => {
                setRetryCountdown((cur) => {
                  if (cur === null || cur <= 1) {
                    if (countdownTimerRef.current !== null) {
                      window.clearInterval(countdownTimerRef.current)
                      countdownTimerRef.current = null
                    }
                    setRateLimited(false)
                    return null
                  }
                  return cur - 1
                })
              }, 1000)
              countdownTimerRef.current = timerId
            }
            return
          }
          // Defensive fallthrough — classifier returned non-rate-limited for 429.
          // Treat as generic error.
          setError('Something went wrong. Try again.')
          return
        }
        // 5xx and any other unexpected status fall here.
        setError('Something went wrong. Try again.')
      } catch {
        setError('Something went wrong. Try again.')
      } finally {
        setSubmitting(false)
      }
    }
    ```
