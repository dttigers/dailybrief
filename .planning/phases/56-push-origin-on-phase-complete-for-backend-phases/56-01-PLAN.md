    ```js
    /**
     * Push phase commits to origin when the completed phase touched any
     * configured deploy target directory.
     *
     * Phase 56: eliminates the 53-04 foot-gun where local main drifted
     * 68 commits ahead of origin and the Mac UI was tested against a
     * stale deployed backend. Behavior:
     *
     *   - No `workflow.deploy_targets` configured -> no-op (return early).
     *   - Deploy targets configured but phase commits touched only
     *     non-target paths (e.g. Mac-only phase) -> no-op. Zero git push.
     *   - At least one phase commit touched a target path -> git push
     *     origin main, loud log line, success.
     *   - git push failure -> THROW. Do not swallow. Phase-complete must
     *     surface this so the user reconciles before marking done.
     *
     * No Railway deploy polling (D-04 defers it). No prompt (D-03).
     */
    function maybePushPhaseCommits(cwd, phaseNum) {
      // D-01: read config.workflow.deploy_targets; defensive on shape
      const cfg = loadConfig(cwd);
      const targets = (cfg && cfg.workflow && Array.isArray(cfg.workflow.deploy_targets))
        ? cfg.workflow.deploy_targets.filter(t => typeof t === 'string' && t.length > 0)
        : [];
      if (targets.length === 0) {
        return { pushed: false, reason: 'no_targets' };
      }

      // Derive phase commit range via commit-message grep.
      // Conventional commits style: feat(56): ..., docs(56-01): ..., fix(56): ...
      // --fixed-strings is NOT used because we want to match both "(56)" and
      // "(56-01)" with one grep AND we need the [^0-9] character class below.
      //
      // Why the trailing `[^0-9]`: without it, the literal substring "(56"
      // would also match future 3-digit phases like "feat(560):" and
      // "feat(561):" — false-positive. The non-digit boundary prevents that.
      // Phase 156 is already safe (the "1" precedes "5"), but any 3-digit
      // phase starting with the current phase number would false-match.
      //
      // The literal "(" must be escaped to "\(" for ERE. The phase number
      // itself is all digits so escapeRegex leaves it unchanged, but we call
      // it anyway for defensiveness. The `[^0-9]` character class is appended
      // AFTER escaping so its regex metachars survive.
      const escapedPhase = escapeRegex(String(phaseNum));
      const grepPattern = `\\(${escapedPhase}[^0-9]`;
      const logResult = execGit(cwd, [
        'log',
        `--grep=${grepPattern}`,
        '--extended-regexp',
        '--pretty=format:%H',
        'main',
      ]);
      if (logResult.exitCode !== 0) {
        // git log failing is a hard error — we cannot reason about the range.
        throw new Error(
          `maybePushPhaseCommits: git log failed (exit ${logResult.exitCode}): ${logResult.stderr}`
        );
      }
      const shas = logResult.stdout.split('\n').map(s => s.trim()).filter(Boolean);
      if (shas.length === 0) {
        return { pushed: false, reason: 'no_matching_commits' };
      }

      // Collect every file path touched by those commits (union across all SHAs).
      const touchedFiles = new Set();
      for (const sha of shas) {
        const showResult = execGit(cwd, ['show', '--name-only', '--pretty=format:', sha]);
        if (showResult.exitCode !== 0) {
          throw new Error(
            `maybePushPhaseCommits: git show ${sha} failed: ${showResult.stderr}`
          );
        }
        for (const line of showResult.stdout.split('\n')) {
          const trimmed = line.trim();
          if (trimmed) touchedFiles.add(trimmed);
        }
      }

      // Directory-prefix match: a deploy_target matches if any touched file
      // starts with `${target}/`. Recursive. Not glob. Per D-01.
      const touchedTargets = new Set();
      for (const file of touchedFiles) {
        for (const target of targets) {
          const prefix = target.endsWith('/') ? target : `${target}/`;
          if (file.startsWith(prefix)) {
            touchedTargets.add(target);
            break;
          }
        }
      }

      if (touchedTargets.size === 0) {
        // Mac-only / non-deploy phase path — the zero-behavior-change case.
        return { pushed: false, reason: 'no_matching_files' };
      }

      // Positive match: push. D-03 — no prompt, loud log, failure = throw.
      const touchedList = [...touchedTargets].sort().join(', ');
      const commitCount = shas.length;
      process.stdout.write(
        `\n◆ Phase ${phaseNum} touched deploy targets: ${touchedList}\n` +
        `◆ Pushing ${commitCount} commit${commitCount === 1 ? '' : 's'} to origin/main — Railway will redeploy\n`
      );
      const pushResult = execGit(cwd, ['push', 'origin', 'main']);
      if (pushResult.exitCode !== 0) {
        // D-03: LOUD error, do not swallow. Throw to block phase-complete success.
        throw new Error(
          `Phase ${phaseNum} push to origin/main FAILED — phase-complete blocked.\n` +
          `Touched deploy targets: ${touchedList}\n` +
          `git push stderr:\n${pushResult.stderr || '(empty)'}\n` +
          `Reconcile manually: git fetch origin && git status`
        );
      }
      process.stdout.write(`◆ Pushed ${commitCount} commit${commitCount === 1 ? '' : 's'} to origin/main successfully\n\n`);
      return {
        pushed: true,
        commit_count: commitCount,
        touched_targets: [...touchedTargets],
      };
    }
    ```

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| user's local repo → origin remote | `git push origin main` runs as a side-effect of a phase-complete call. This is the only new network egress this phase introduces. |
| `.planning/config.json` → phase.cjs runtime | The `workflow.deploy_targets` list is read from an in-repo, user-controlled JSON file and used to decide whether to push. |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-56-01 | Tampering | `.planning/config.json` deploy_targets | accept | File is in-repo and reviewable via git. Solo-dev workflow — user is the only writer. A malicious edit that sets `deploy_targets: [""]` or `["."]` would make EVERY phase-complete push; mitigation is that such an edit is visible in `git diff`. Code defense: Task 3 B2 uses strict `file.startsWith(prefix)` with `prefix = target + '/'`, so an empty string still requires `file.startsWith('/')` — no file matches, no push. Verified: empty-string target in deploy_targets cannot trigger a push. |
| T-56-02 | Repudiation / Information Disclosure | Unintended commits in push | accept | `git push origin main` (no `--force`) rejects non-fast-forward pushes naturally. If local history was rewritten or contains commits the user did not intend to publish, git's own non-FF protection triggers a loud failure — which per D-03 blocks phase-complete. Mitigation is therefore automatic via git's refusal; no extra pre-check needed. |
| T-56-03 | Denial of Service | Phase-complete blocked by network failure | accept | If origin is unreachable (no network, auth expired, etc.), push fails loudly and phase-complete halts. This is the DESIGNED behavior (D-03). User must reconcile and re-run. Not a bug. No mitigation — this is the feature. |
| T-56-04 | Elevation of Privilege | New credential surface | mitigate | Zero new auth surface introduced. The code calls `git push` via existing `execGit` spawnSync, which inherits the user's configured git credential helper (same credentials phase-complete's other git calls already use). No new env vars, no new secret storage, no new CLI dependency. |
| T-56-05 | Spoofing | Malicious phase number as arg | accept | `phaseNum` is passed from the already-validated `cmdPhaseComplete` arg path (normalizePhaseName + findPhaseInternal upstream). Passed to `git log --grep="(${phaseNum}"`. escapeRegex is applied before the grep. No shell interpolation — spawnSync with arg array eliminates injection. |
| T-56-06 | Tampering | Commit-message grep miss | mitigate | If a phase's commits used a non-standard subject format that does NOT contain `(${phaseNum}`, the grep returns no SHAs and the function silently reports `no_matching_commits` — meaning a real vigil-core change could NOT be detected and NOT be pushed. Mitigation: document in the helper's JSDoc that commits MUST follow `type(phaseNum): subject` conventional commits format (they already do — this is project policy). Residual risk: if the policy breaks, push silently skips. Acceptable for v1; revisit if it ever bites. |
