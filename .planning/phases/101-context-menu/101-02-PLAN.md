```tsx
import { useCallback, useEffect, useLayoutEffect, useRef, useState } from 'react'
import { createPortal } from 'react-dom'
import type { ThoughtApiResponse, ProjectApiResponse } from '../api/client'
import { CATEGORIES } from '../constants/categories'

export interface ContextMenuProps {
  anchor: { x: number; y: number }
  thought: ThoughtApiResponse
  projects: ProjectApiResponse[]
  openedVia: 'mouse' | 'touch'
  onClose: () => void
  onStartEdit: () => void
  onRetriage: (id: number) => void
  onMoveToCategory: (id: number, category: string) => void
  onAssignProject: (id: number, projectId: number) => void
  onDelete: (id: number) => void
}

type View = 'root' | 'categories' | 'projects'

const ALPHABETICAL_CATEGORIES = [...CATEGORIES].sort() as readonly string[]

export default function ContextMenu(props: ContextMenuProps) {
  const { anchor, thought, projects, openedVia, onClose } = props
  const [view, setView] = useState<View>('root')
  const [desktopSubmenu, setDesktopSubmenu] = useState<'categories' | 'projects' | null>(null)
  const menuRef = useRef<HTMLDivElement>(null)
  const [adjusted, setAdjusted] = useState(anchor)

  // D-08: viewport-overflow positioning — run AFTER menu renders so we can measure.
  useLayoutEffect(() => {
    if (!menuRef.current) return
    const rect = menuRef.current.getBoundingClientRect()
    const vw = window.innerWidth
    const vh = window.innerHeight
    let { x, y } = anchor
    if (x + rect.width > vw - 8) x = Math.max(8, vw - rect.width - 8)
    if (y + rect.height > vh - 8) y = Math.max(8, y - rect.height)
    x = Math.max(8, x)
    y = Math.max(8, y)
    setAdjusted({ x, y })
  }, [anchor, view])  // re-measure when view changes (mobile in-place replace can change height)

  // D-07: close on Escape, outside pointerdown, scroll, resize.
  useEffect(() => {
    const onKey = (e: KeyboardEvent) => { if (e.key === 'Escape') onClose() }
    const onScroll = () => onClose()
    const onResize = () => onClose()
    const onDown = (e: PointerEvent) => {
      if (menuRef.current && !menuRef.current.contains(e.target as Node)) onClose()
    }
    window.addEventListener('keydown', onKey)
    window.addEventListener('scroll', onScroll, true)  // capture — nested scroll containers (Pattern 4)
    window.addEventListener('resize', onResize)
    window.addEventListener('pointerdown', onDown)
    return () => {
      window.removeEventListener('keydown', onKey)
      window.removeEventListener('scroll', onScroll, true)
      window.removeEventListener('resize', onResize)
      window.removeEventListener('pointerdown', onDown)
    }
  }, [onClose])

  // ... view rendering + submenu rendering below
}
```

**Rendering rules:**

Root view (`view === 'root'`) items in order — each with `role="menuitem"`:
1. `<button>Edit</button>` — onClick: `props.onStartEdit(); onClose()`
2. `<button>Re-triage</button>` — onClick: `props.onRetriage(thought.id); onClose()`
3. `<button>Move to category <span>→</span></button>` — desktop: `onMouseEnter={() => setDesktopSubmenu('categories')}`; mobile (openedVia='touch'): `onClick={() => setView('categories')}`
4. `<button>Add to project <span>→</span></button>` — same shape as Move
5. `<div className="border-t border-gray-700 my-1" />` — separator per D-09
6. `<button className="... text-red-400 hover:bg-red-500/10">Delete</button>` — onClick: `props.onDelete(thought.id); onClose()`

Mobile `view === 'categories'` renders:
- Header button: `← Categories` (onClick returns to root: `setView('root')`)
- `ALPHABETICAL_CATEGORIES.map(cat => <button role="menuitem">cat {cat === thought.category && <span className="text-teal-400">✓</span>}</button>)` — onClick: `props.onMoveToCategory(thought.id, cat); onClose()`

Mobile `view === 'projects'` renders:
- Header button: `← Projects`
- If `projects.length === 0`: `<p className="text-sm text-gray-400 text-center px-4 py-2">No projects yet. Create one on the Projects tab.</p>`
- Else `[...projects].sort((a,b)=>a.name.localeCompare(b.name)).map(p => <button role="menuitem">p.name {p.id === thought.projectId && ✓}</button>)` — onClick: `props.onAssignProject(thought.id, p.id); onClose()`

Desktop submenu (when `openedVia === 'mouse'` and `desktopSubmenu !== null`):
- Render a second portal (or sibling div) with surface classes `bg-gray-800 border border-gray-700`
- Position: offset to the right of parent menu rect (e.g. `left: parentRect.right + 4`, same `top`). Keep it simple — use a `ref` on root to read rect, or store parent rect in state.
- Contents: same category or project list as mobile view but without the back header.
- Hovering away (`onMouseLeave`) on the parent row closes the submenu after a 0ms delay (UI-SPEC §Interaction — no hover-intent in this phase).
