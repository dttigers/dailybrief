interface WorkOrdersDeps {
  dbAvailable: boolean;
  // Used when wo.clientCaptureId is non-null. Returns the persisted row and whether
  // this call inserted (isNew:true) or matched an existing (userId, clientCaptureId)
  // row (isNew:false). The partial unique index from 129-01 is the DB-side race
  // safety net; this fn is the application-layer mirror.
  dbInsertOrGet: (input: SanitizedWorkOrder) => Promise<{ row: WorkOrderRow; isNew: boolean }>;
  // Used when wo.clientCaptureId is null/omitted (legacy/CLI path). Routes through
  // the existing ON CONFLICT (case_number) DO UPDATE upsert — unchanged semantics.
  dbUpsertLegacy: (input: SanitizedWorkOrder) => Promise<{ row: WorkOrderRow }>;
}

}

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Extension POST body → route sanitizer | Untrusted JSON; mass-assignment risk on any pass-through. |
| Route → DB write | userId MUST come from authenticated middleware context, NOT from the request body. |
